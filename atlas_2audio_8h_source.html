22/**
23 * @brief Component that provides audio playback capabilities to an object. It
24 * can be attach to any object in the scene and allows for playing, pausing and
25 * stopping audio.
26 *
27 * \note Finewave will be required for this component to work. This component
28 * under the hood uses Finewave's AudioSource class to manage audio playback.
29 *
30 * \subsection audio-player-example Example
31 * ```cpp
32 * // Create an AudioPlayer component
33 * auto audioPlayer = std::make_shared<AudioPlayer>();
34 * // Initialize the audio player
35 * audioPlayer->init();
36 * // Set the audio source from a file
37 * audioPlayer->setSource(Workspace::get().getResource("AudioResource"));
38 * // Play the audio
39 * audioPlayer->play();
40 * ```
41 */
42class AudioPlayer : public Component {
43 public:
44 /**
45 * @brief Construct a new empty AudioPlayer object
46 *
47 */
48 AudioPlayer() : sourceInitialized(false) {}

51 * @brief Destroy the Audio Player object and release resources.

54 AudioPlayer(const AudioPlayer &) = delete;

58 * @return AudioPlayer&
59 */
60 AudioPlayer &operator=(const AudioPlayer &) = delete;

62 /**
63 * @brief Move constructor for AudioPlayer.

65 * @param other The AudioPlayer instance to move from.

88 /**
89 * @brief Initialize the audio player and prepare it for playback.

126 /**
127 * @brief Set the source from where the audio will be played.
128 *
129 * @param sourceResource The resource containing the audio file.
130 */
131 inline void setSource(Resource sourceResource) {
132 ensureSourceInitialized();
133 source->fromFile(sourceResource);
134 }
135
136 /**
137 * @brief Update the audio player state. This method is called every frame.
138 *
139 * @param dt The delta time since the last update.
140 */
141 inline void update(float dt) override {
142 ensureSourceInitialized();
143 Window *window = Window::mainWindow;
144 Camera *mainCamera = window->getCamera();
145
146 if (mainCamera) {
147 Position3d cameraPos = mainCamera->position;
148 Normal3d cameraFront = mainCamera->getFrontVector();
149 window->audioEngine->setListenerPosition(cameraPos);
150 window->audioEngine->setListenerOrientation(
151 cameraFront, Normal3d{0.0f, 1.0f, 0.0f});
152 window->audioEngine->setListenerVelocity(mainCamera->getVelocity());
153 }
154
155 source->setPosition(this->object->getPosition());
156 }
157
158 /**
159 * @brief Set the position of the audio source in 3D space. \warning This is
160 * used when spatialization is on.
161 *
162 * @param position The new position of the audio source.
163 */
164 inline void setPosition(Position3d position) {
165 ensureSourceInitialized();
166 source->setPosition(position);
167 }
168
169 /**
170 * @brief Enable spatialization for the audio source, making it 3D and
171 * affected by the listener's position.
172 *
173 */
174 inline void useSpatialization() {
175 ensureSourceInitialized();
176 source->useSpatialization();
177 }
178
179 /**
180 * @brief Disable spatialization for the audio source, making it play
181 * uniformly regardless of the listener's position.
182 *
183 */
184 inline void disableSpatialization() {
185 ensureSourceInitialized();
186 source->disableSpatialization();
187 }
188
189 /**
190 * @brief Get the audio source. This allows for advanced manipulation of the
191 * audio playback.
192 *
193 */
194 std::unique_ptr<AudioSource> source;
195
196 private:
197 bool sourceInitialized;
198
199 inline void ensureSourceInitialized() {
200 if (!sourceInitialized) {
201 source = std::make_unique<AudioSource>();
202 sourceInitialized = true;
203 }
204 }
205};
