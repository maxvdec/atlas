58/**
59 * @brief Base class for all post-processing effects that can be applied to a
60 * render target. Effects are applied via shader uniforms.
61 *
62 */
63class Effect {
64 public:
65 /**
66 * @brief The type of effect this instance represents.
67 *
68 */
69 RenderTargetEffect type;
70 /**
71 * @brief Constructs a new Effect object.
72 *
73 * @param t The type of effect.
74 */
75 Effect(RenderTargetEffect t) : type(t) {}
76 /**
77 * @brief Applies the effect's parameters to the shader program.
78 *
79 * @param program The shader program to apply the effect to.
80 * @param index The index of the effect in the effect array.
81 */
82 virtual void applyToProgram(ShaderProgram &program, int index) {};
83};

163 /**
164 * @brief Constructs a new Blur effect.
165 *
166 * @param magnitude The blur radius. Higher values create stronger blur.

168 Blur(float magnitude = 16.0)
169 : Effect(RenderTargetEffect::Blur), magnitude(magnitude) {}
170 /**

186 program.setUniform1f("EffectFloat1[" + std::to_string((int)index) + "]",

214/**
215 * @brief Structure containing parameters for color correction post-processing.
216 * Allows fine-tuning of exposure, contrast, saturation, gamma, temperature,
217 * and tint.
218 *
219 */
220struct ColorCorrectionParameters {
221 /**
222 * @brief Exposure adjustment. Positive values brighten, negative values
223 * darken.
224 *
225 */
226 float exposure = 0.0;
227 /**
228 * @brief Contrast adjustment. Values > 1.0 increase contrast, < 1.0
229 * decrease it.
230 *
231 */
232 float contrast = 1.0;
233 /**
234 * @brief Saturation adjustment. 1.0 is normal, 0.0 is grayscale, > 1.0
235 * increases saturation.
236 *
237 */
238 float saturation = 1.0;
239 /**
240 * @brief Gamma correction value. Typically around 1.0 to 2.2.
241 *
242 */
243 float gamma = 1.0;
244 /**
245 * @brief Temperature adjustment. Positive values add warmth (red),
246 * negative values add coolness (blue).
247 *
248 */
249 float temperature = 0.0;
250 /**
251 * @brief Tint adjustment. Positive values add green, negative values add
252 * magenta.
253 *
254 */
255 float tint = 0.0;
256};
257
258/**
259 * @brief Post-processing effect that applies comprehensive color correction to
260 * the rendered image, including exposure, contrast, saturation, and color
261 * temperature adjustments.
262 *
263 * \subsection color-correction-example Example
264 * ```cpp
265 * // Create a render target
266 * RenderTarget renderTarget(window);
267 * // Define color correction parameters
268 * ColorCorrectionParameters params;
269 * params.exposure = 0.5f; // Increase brightness
270 * params.contrast = 1.2f; // Increase contrast slightly
271 * params.saturation = 1.1f; // Boost saturation
272 * params.temperature = 0.1f; // Add warmth
273 * // Create and add the color correction effect
274 * auto colorCorrection = ColorCorrection::create(params);
275 * renderTarget.addEffect(colorCorrection);
276 * ```
277 *
278 */
279class ColorCorrection : public Effect {
280 public:
281 /**
282 * @brief The color correction parameters to apply.
283 *
284 */
285 ColorCorrectionParameters params;
286
287 /**
288 * @brief Constructs a new ColorCorrection effect.
289 *
290 * @param p The color correction parameters to use.
291 */
292 ColorCorrection(ColorCorrectionParameters p = {})
293 : Effect(RenderTargetEffect::ColorCorrection), params(p) {}
294 /**
295 * @brief Creates a shared pointer to a ColorCorrection effect.
296 *
297 * @param p The color correction parameters to use.
298 * @return (std::shared_ptr<ColorCorrection>) The created color correction
299 * effect.
300 */
301 static std::shared_ptr<ColorCorrection>
302 create(ColorCorrectionParameters p = {}) {
303 return std::make_shared<ColorCorrection>(p);
304 }
305 /**
306 * @brief Applies all color correction parameters to the shader program.
307 *
308 * @param program The shader program to apply the effect to.
309 * @param index The index of the effect in the effect array.
310 */
311 void applyToProgram(ShaderProgram &program, int index) override {
312 program.setUniform1f("EffectFloat1[" + std::to_string((int)index) + "]",
313 params.exposure);
314 program.setUniform1f("EffectFloat2[" + std::to_string((int)index) + "]",
315 params.contrast);
316 program.setUniform1f("EffectFloat3[" + std::to_string((int)index) + "]",
317 params.saturation);
318 program.setUniform1f("EffectFloat4[" + std::to_string((int)index) + "]",
319 params.gamma);
320 program.setUniform1f("EffectFloat5[" + std::to_string((int)index) + "]",
321 params.temperature);
322 program.setUniform1f("EffectFloat6[" + std::to_string((int)index) + "]",
323 params.tint);
324 }
325};
