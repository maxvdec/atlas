54/**
55 * @brief Base class for all post-processing effects that can be applied to a
56 * render target. Effects are applied via shader uniforms.
57 *
58 */
59class Effect {
60 public:
61 /**
62 * @brief The type of effect this instance represents.
63 *
64 */
65 RenderTargetEffect type;
66 /**
67 * @brief Constructs a new Effect object.
68 *
69 * @param t The type of effect.
70 */
71 Effect(RenderTargetEffect t) : type(t) {}
72 /**
73 * @brief Applies the effect's parameters to the shader program.
74 *
75 * @param program The shader program to apply the effect to.
76 * @param index The index of the effect in the effect array.
77 */
78 virtual void applyToProgram(ShaderProgram &program, int index) {};
79};

147/**
148 * @brief Post-processing effect that applies a Gaussian blur to the rendered
149 * image. The blur magnitude can be controlled.
150 *
151 */
152class Blur : public Effect {
153 public:
154 /**
155 * @brief The magnitude (radius) of the blur effect.
156 *
157 */
158 float magnitude = 16.0;
159 /**
160 * @brief Constructs a new Blur effect.
161 *
162 * @param magnitude The blur radius. Higher values create stronger blur.
163 */
164 Blur(float magnitude = 16.0)
165 : Effect(RenderTargetEffect::Blur), magnitude(magnitude) {}
166 /**
167 * @brief Creates a shared pointer to a Blur effect.
168 *
169 * @param magnitude The blur radius.
170 * @return (std::shared_ptr<Blur>) The created blur effect.
171 */
172 static std::shared_ptr<Blur> create(float magnitude = 16.0) {
173 return std::make_shared<Blur>(magnitude);
174 }
175 /**
176 * @brief Applies the blur magnitude to the shader program.
177 *
178 * @param program The shader program to apply the effect to.
179 * @param index The index of the effect in the effect array.
180 */
181 void applyToProgram(ShaderProgram &program, int index) override {
182 program.setUniform1f("EffectFloat1[" + std::to_string((int)index) + "]",
183 magnitude);
184 }
185};

210/**
211 * @brief Structure containing parameters for color correction post-processing.
212 * Allows fine-tuning of exposure, contrast, saturation, gamma, temperature,
213 * and tint.
214 *
215 */
216struct ColorCorrectionParameters {
217 /**
218 * @brief Exposure adjustment. Positive values brighten, negative values
219 * darken.
220 *
221 */
222 float exposure = 0.0;
223 /**
224 * @brief Contrast adjustment. Values > 1.0 increase contrast, < 1.0
225 * decrease it.
226 *
227 */
228 float contrast = 1.0;
229 /**
230 * @brief Saturation adjustment. 1.0 is normal, 0.0 is grayscale, > 1.0
231 * increases saturation.
232 *
233 */
234 float saturation = 1.0;
235 /**
236 * @brief Gamma correction value. Typically around 1.0 to 2.2.
237 *
238 */
239 float gamma = 1.0;
240 /**
241 * @brief Temperature adjustment. Positive values add warmth (red),
242 * negative values add coolness (blue).
243 *
244 */
245 float temperature = 0.0;
246 /**
247 * @brief Tint adjustment. Positive values add green, negative values add
248 * magenta.
249 *
250 */
251 float tint = 0.0;
252};
253
254/**
255 * @brief Post-processing effect that applies comprehensive color correction to
256 * the rendered image, including exposure, contrast, saturation, and color
257 * temperature adjustments.
258 *
259 * \subsection color-correction-example Example
260 * ```cpp
261 * // Create a render target
262 * RenderTarget renderTarget(window);
263 * // Define color correction parameters
264 * ColorCorrectionParameters params;
265 * params.exposure = 0.5f; // Increase brightness
266 * params.contrast = 1.2f; // Increase contrast slightly
267 * params.saturation = 1.1f; // Boost saturation
268 * params.temperature = 0.1f; // Add warmth
269 * // Create and add the color correction effect
270 * auto colorCorrection = ColorCorrection::create(params);
271 * renderTarget.addEffect(colorCorrection);
272 * ```
273 *
274 */
275class ColorCorrection : public Effect {
276 public:
277 /**
278 * @brief The color correction parameters to apply.
279 *
280 */
281 ColorCorrectionParameters params;
282
283 /**
284 * @brief Constructs a new ColorCorrection effect.
285 *
286 * @param p The color correction parameters to use.
287 */
288 ColorCorrection(ColorCorrectionParameters p = {})
289 : Effect(RenderTargetEffect::ColorCorrection), params(p) {}
290 /**
291 * @brief Creates a shared pointer to a ColorCorrection effect.
292 *
293 * @param p The color correction parameters to use.
294 * @return (std::shared_ptr<ColorCorrection>) The created color correction
295 * effect.
296 */
297 static std::shared_ptr<ColorCorrection>
298 create(ColorCorrectionParameters p = {}) {
299 return std::make_shared<ColorCorrection>(p);
300 }
301 /**
302 * @brief Applies all color correction parameters to the shader program.
303 *
304 * @param program The shader program to apply the effect to.
305 * @param index The index of the effect in the effect array.
306 */
307 void applyToProgram(ShaderProgram &program, int index) override {
308 program.setUniform1f("EffectFloat1[" + std::to_string((int)index) + "]",
309 params.exposure);
310 program.setUniform1f("EffectFloat2[" + std::to_string((int)index) + "]",
311 params.contrast);
312 program.setUniform1f("EffectFloat3[" + std::to_string((int)index) + "]",
313 params.saturation);
314 program.setUniform1f("EffectFloat4[" + std::to_string((int)index) + "]",
315 params.gamma);
316 program.setUniform1f("EffectFloat5[" + std::to_string((int)index) + "]",
317 params.temperature);
318 program.setUniform1f("EffectFloat6[" + std::to_string((int)index) + "]",
319 params.tint);
320 }
321};
