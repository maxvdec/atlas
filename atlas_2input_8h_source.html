225inline SDL_GamepadAxis atlasToSDLGamepadAxis(int axisIndex) {
226 switch (axisIndex) {
227 case CONTROLLER_AXIS_LEFT_X:
228 return SDL_GAMEPAD_AXIS_LEFTX;
229 case CONTROLLER_AXIS_LEFT_Y:
230 return SDL_GAMEPAD_AXIS_LEFTY;
231 case CONTROLLER_AXIS_RIGHT_X:
232 return SDL_GAMEPAD_AXIS_RIGHTX;
233 case CONTROLLER_AXIS_RIGHT_Y:
234 return SDL_GAMEPAD_AXIS_RIGHTY;
235 case CONTROLLER_AXIS_LEFT_TRIGGER:
236 return SDL_GAMEPAD_AXIS_LEFT_TRIGGER;
237 case CONTROLLER_AXIS_RIGHT_TRIGGER:
238 return SDL_GAMEPAD_AXIS_RIGHT_TRIGGER;

438/**
439 * @brief Abstract class that declares some object that can recieve input and
440 * react to the changes in it.
441 * \subsection interactive-example Example
442 * ```cpp
443 * class PlayerController : public Interactive {
444 * public:
445 * void onKeyPress(Key key, float deltaTime) override {
446 * if (key == Key::W) {
447 * // Move forward
448 * }
449 * }
450 * void onMouseMove(MousePacket data, float deltaTime) override {
451 * // Look around based on mouse movement
452 * }
453 * };
454 * ```
455 *
456 */
457class Interactive {
458 public:
459 virtual ~Interactive() = default;
460 /**
461 * @brief Function that is called automatically when a key is pressed.
462 *
463 * @param key The key that was pressed.
464 * @param deltaTime The time elapsed since the last frame.
465 */
466 virtual void onKeyPress([[maybe_unused]] Key key,
467 [[maybe_unused]] float deltaTime) {};
468 /**
469 * @brief Function that is called automatically when a key is released.
470 *
471 * @param key The key that was released.
472 * @param deltaTime The time elapsed since the last frame.
473 */
474 virtual void onKeyRelease([[maybe_unused]] Key key,
475 [[maybe_unused]] float deltaTime) {};
476 /**
477 * @brief Function that is called automatically when the mouse is moved.
478 *
479 * @param data The mouse movement data. See \ref MousePacket.
480 * @param deltaTime The time elapsed since the last frame.
481 */
482 virtual void onMouseMove([[maybe_unused]] MousePacket data,
483 [[maybe_unused]] float deltaTime) {};
484 /**
485 * @brief Function that is called automatically when a mouse button is
486 * pressed.
487 *
488 * @param button The mouse button that was pressed.
489 * @param deltaTime The time elapsed since the last frame.
490 */
491 virtual void onMouseButtonPress([[maybe_unused]] MouseButton button,
492 [[maybe_unused]] float deltaTime) {};
493 /**
494 * @brief Function that is called automatically when the mouse is scrolled.
495 *
496 * @param data The mouse scroll data. See \ref MouseScrollPacket.
497 * @param deltaTime The time elapsed since the last frame.
498 */
499 virtual void onMouseScroll([[maybe_unused]] MouseScrollPacket data,
500 [[maybe_unused]] float deltaTime) {};
501 /**
502 * @brief Function that is called automatically at each frame.
503 *
504 * @param deltaTime The time elapsed since the last frame.
505 */
506 virtual void atEachFrame([[maybe_unused]] float deltaTime) {};
507
508 /**
509 * @brief The last mouse data received. Updated automatically by the \ref
510 * Window class.
511 *
512 */
513 MousePacket lastMouseData = MousePacket();
514};
