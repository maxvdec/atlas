182/**
183 * @brief Manages day-night cycle, sky colors, lights, and volumetric weather.
184 *
185 * \subsection atmosphere-example Example
186 * ```cpp
187 * // Create atmosphere with day-night cycle
188 * Atmosphere atmosphere;
189 * atmosphere.timeOfDay = 14.5f; // 2:30 PM
190 * atmosphere.secondsPerHour = 120.0f; // Fast time: 2 minutes per hour
191 * atmosphere.enable();
192 *
193 * // Configure volumetric clouds
194 * Clouds clouds(4, 6); // Worley frequency and divisions
195 * clouds.position = Position3d(0.0f, 100.0f, 0.0f);
196 * clouds.size = Size3d(500.0f, 80.0f, 500.0f);
197 * clouds.density = 0.5f;
198 * clouds.absorption = 1.2f;
199 * clouds.scattering = 0.9f;
200 * clouds.wind = Magnitude3d(0.05f, 0.0f, 0.02f);
201 * clouds.primaryStepCount = 16; // Quality vs performance
202 *
203 * // Set up dynamic weather
204 * atmosphere.enableWeather();
205 * atmosphere.weatherDelegate = [](ViewInformation viewInfo) {
206 * WeatherState weather;
207 * weather.condition = WeatherCondition::Rain;
208 * weather.intensity = 0.7f;
209 * weather.wind = Magnitude3d(0.1f, -0.5f, 0.05f);
210 * return weather;
211 * };
212 *
213 * // Update each frame
214 * atmosphere.update(deltaTime);
215 * ```
216 *
217 */
