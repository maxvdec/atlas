136/**
137 * @brief Triangle mesh collider defined by indexed geometry.
138 *
139 * \warning Mesh colliders are typically more expensive than primitives.
140 */
141class MeshCollider : public Collider {
142 public:
143 std::vector<Position3d> vertices;
144 std::vector<uint32_t> indices;
145
146 float getMinExtent() const override {
147 float minExtent = std::numeric_limits<float>::max();
148 for (const auto &vert : vertices) {
149 minExtent = std::min({minExtent, vert.x, vert.y, vert.z});
150 }
151 return minExtent;
152 }
153
154 MeshCollider(const std::vector<Position3d> &vertices,
155 const std::vector<uint32_t> &indices);
