134/**
135 * @brief Triangle mesh collider defined by indexed geometry.
136 *
137 * \warning Mesh colliders are typically more expensive than primitives.
138 */
139class MeshCollider : public Collider {
140 public:
141 std::vector<Position3d> vertices;
142 std::vector<uint32_t> indices;
143
144 float getMinExtent() const override {
145 float minExtent = std::numeric_limits<float>::max();
146 for (const auto &vert : vertices) {
147 minExtent = std::min({minExtent, vert.x, vert.y, vert.z});
148 }
149 return minExtent;
150 }
151
152 MeshCollider(const std::vector<Position3d> &vertices,
153 const std::vector<uint32_t> &indices);
154#ifndef BEZEL_NATIVE
155 JPH::RefConst<JPH::Shape> getJoltShape() const override;
