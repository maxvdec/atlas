150 /**
151 * @brief Applies mass to the body, setting the inverse mass.
152 *
153 * @param mass The mass to apply. If <= 0, creates infinite mass.
154 */
155 inline void applyMass(float mass) {
156 if (mass <= 0.0f) {
157 invMass = 0.0f; // Infinite mass
158 } else {
159 invMass = 1.0f / mass;
160 }
161 }
162
163 /**
164 * @brief Gets the mass of the body.
165 *
166 * @return (float) The mass of the body, or INFINITY for infinite mass.
167 */
168 inline float getMass() const {
169 if (invMass == 0.0f) {
170 return INFINITY;
171 }
172 return 1.0f / invMass;
173 }
