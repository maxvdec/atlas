58/**

66 * // Set the body's position
67 * body->position = {0.0, 0.0, 0.0};
68 * // Apply a mass of 2.0 units

149 /**
150 * @brief Applies mass to the body, setting the inverse mass.
151 *
152 * @param mass The mass to apply. If <= 0, creates infinite mass.
153 */
154 inline void applyMass(float mass) {
155 if (mass <= 0.0f) {
156 invMass = 0.0f; // Infinite mass
157 } else {
158 invMass = 1.0f / mass;
159 }
160 }
161
162 /**
163 * @brief Gets the mass of the body.
164 *
165 * @return (float) The mass of the body, or INFINITY for infinite mass.
166 */
167 inline float getMass() const {
168 if (invMass == 0.0f) {
169 return INFINITY;
170 }
171 return 1.0f / invMass;
172 }
