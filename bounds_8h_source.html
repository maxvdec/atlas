34 * if (objectBounds.doesIntersect(otherBounds)) {

36 * }

38 * // Get bounds dimensions
39 * float width = objectBounds.widthX();
40 * float height = objectBounds.widthY();
41 * float depth = objectBounds.widthZ();

43 * // Broad-phase collision detection for physics simulation
44 * std::vector<std::shared_ptr<Body>> bodies = {...};

47 *
48 * // Process potential collisions

135/**
136 * @brief Structure representing a pair of potentially colliding bodies.
137 *
138 */
139struct CollisionPair {
140 int a;
141 int b;
142
143 inline bool operator==(const CollisionPair &other) const {
144 return (a == other.a && b == other.b) || (a == other.b && b == other.a);
145 }
146
147 inline bool operator!=(const CollisionPair &other) const {
148 return !(*this == other);
149 }
150};
