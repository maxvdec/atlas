103/**
104 * @brief Structure representing a pair of potentially colliding bodies.
105 *
106 */
107struct CollisionPair {
108 int a;
109 int b;
110
111 inline bool operator==(const CollisionPair &other) const {
112 return (a == other.a && b == other.b) || (a == other.b && b == other.a);
113 }
114
115 inline bool operator!=(const CollisionPair &other) const {
116 return !(*this == other);
117 }
118};
