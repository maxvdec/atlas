19 * @brief Class representing a camera in 3D space, capable of generating view
20 * and projection matrices. It also has built-in controls for moving and looking
21 * with native support for both perspective and orthographic projections.
22 *
23 * \subsection example-camera-control Example of camera control
24 * ```cpp
25 * // Create a camera instance
26 * Camera camera;
27 * // Set the camera position
28 * camera.setPosition({0.0f, 0.0f, 5.0f});
29 * // Make the camera look at the origin
30 * camera.lookAt({0.0f, 0.0f, 0.0f});
31 * // On each frame, update the camera with input from the window
32 * camera.update(window);
33 * // On each mouse movement, update the look direction
34 * camera.updateLook(window, mouseMovement);

161 /**
162 * @brief Get the front vector of the camera, representing the direction it
163 * is facing.
164 *
165 * @return (Normal3d) The front vector of the camera.
166 */
167 inline Normal3d getFrontVector() const {
168 Normal3d front;
169 front.x = cos(glm::radians(yaw)) * cos(glm::radians(pitch));
170 front.y = sin(glm::radians(pitch));
171 front.z = sin(glm::radians(yaw)) * cos(glm::radians(pitch));
172 return Normal3d::fromGlm(glm::normalize(front.toGlm()));
173 }
174
175 /**

177 * movement speed.
178 *
179 * @return (Magnitude3d) The velocity of the camera.
180 */
181 inline Magnitude3d getVelocity() const {
182 return getFrontVector() * movementSpeed;
183 }
