18/**
19 * @brief Class representing a camera in 3D space, capable of generating view

24 * ```cpp
25 * // Create a camera instance
26 * Camera camera;
27 * // Set the camera position
28 * camera.setPosition({0.0f, 0.0f, 5.0f});

30 * camera.lookAt({0.0f, 0.0f, 0.0f});

174 /**
175 * @brief Get the front vector of the camera, representing the direction it
176 * is facing.
177 *
178 * @return (Normal3d) The front vector of the camera.
179 */
180 inline Normal3d getFrontVector() const {
181 Normal3d front;
182 front.x = cos(glm::radians(yaw)) * cos(glm::radians(pitch));
183 front.y = sin(glm::radians(pitch));
184 front.z = sin(glm::radians(yaw)) * cos(glm::radians(pitch));
185 return Normal3d::fromGlm(glm::normalize(front.toGlm()));
186 }
187
188 /**
189 * @brief Get the velocity of the camera based on its front vector and
190 * movement speed.
191 *
192 * @return (Magnitude3d) The velocity of the camera.
193 */
194 inline Magnitude3d getVelocity() const {
195 return getFrontVector() * movementSpeed;
196 }
