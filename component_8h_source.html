25/**
26 * @brief Behavior or property that can be attached to a \ref GameObject to
27 * extend its capabilities.
28 *
29 * \subsection example-component Example of a component
30 * ```cpp
31 * // Define a custom component by inheriting from Component
32 * class CustomComponent : public Component {
33 * public:
34 * void init() override {
35 * // Custom initialization code here
36 * }
37 * void update(float deltaTime) override {
38 * // Custom update code here
39 * }
40 * };
41 * ```
42 */
43class Component {
44 public:
45 /**
46 * @brief Initializes the component. This method is called once when the
47 * component is added to a GameObject.
48 *
49 */
50 virtual void init() {}
51 /**
52 * @brief Updates the component each frame. This method is called every
53 * frame before rendering.
54 *
55 * @param deltaTime The time elapsed since the last frame.
56 */
57 virtual void update(float deltaTime) {}
58 /**
59 * @brief Performs changes when the GameObject's view matrix is updated.
60 *
61 * @param view The new view matrix.
62 */
63 virtual void setViewMatrix(const glm::mat4 &view) {}
64 /**
65 * @brief Performs changes when the GameObject's projection matrix is
66 * updated.
67 *
68 * @param projection The new projection matrix.
69 */
70 virtual void setProjectionMatrix(const glm::mat4 &projection) {}
71 /**
72 * @brief Gets the window associated with the component's GameObject.
73 *
74 * @return (Window*) The window instance.
75 */
76 Window *getWindow();
77
78 /**
79 * @brief Construct a new Component object
80 *
81 */
82 Component() = default;
83
84 /**
85 * @brief Gets the GameObject associated with the component.
86 *
87 */
88 GameObject *object = nullptr;
89 /**
90 * @brief Gets the Body associated with the component.
91 *
92 */
93 Body *body = nullptr;
94};

96/**
97 * @brief Component that is specifically tied to a certain type of \ref
98 * GameObject. It cannot be added to objects that do not inherit from the
99 * specified type.
100 *
101 * @tparam T The type of GameObject the component is tied to.
102 */
103template <typename T>
104 requires std::is_base_of_v<GameObject, T>
105class TraitComponent : public Component {
106 public:
107 /**
108 * @brief Initializes the component. This method is called once when the
109 * component is added to a GameObject.
110 *
111 */
112 virtual void init() override {};
113 /**
114 * @brief Updates the component each frame. This method is called every
115 * frame before rendering.
116 *
117 * @param deltaTime The time elapsed since the last frame.
118 */
119 void update(float deltaTime) override {
120 if (typedObject != nullptr) {
121 updateComponent(typedObject);
122 }
123 }
124 /**
125 * @brief Updates the component with a typed reference to the GameObject it
126 * is bound to.
127 *
128 * @param object A pointer to the typed GameObject.
129 */
130 virtual void updateComponent(T *object) {}
131 /**
132 * @brief Sets the typed object reference for the component.
133 *
134 * @param obj A pointer to the typed GameObject.
135 */
136 inline void setTypedObject(T *obj) { typedObject = obj; }
137
138 inline T *getObject() { return typedObject; }
139
140 private:
141 T *typedObject = nullptr;
142};

230 /**
231 * @brief Adds a component to the object.
232 *
233 * @tparam T The type of component to add.
234 * @param existing The component instance to add. \warning It must be
235 * long-lived. This means that declaring it as a class property is a good
236 * idea.
237 */
238 template <typename T>
239 requires std::is_base_of_v<Component, T>
240 void addComponent(T &&existing) {
241 std::shared_ptr<T> component =
242 std::make_shared<T>(std::forward<T>(existing));
243 component->object = this;
244 component->body = this->body.get();
245 components.push_back(component);
246 }
247
248 /**
249 * @brief Adds a trait component to the object. A trait component is a
250 * component that is tied to a specific type of GameObject.
251 *
252 * @tparam U The type of GameObject the trait component is tied to.
253 * @tparam T The type of trait component to add.
254 * @param existing The trait component instance to add. \warning It must be
255 * long-lived. This means that declaring it as a class property is a good
256 * idea.
257 */
258 template <typename U, typename T>
259 requires std::is_base_of_v<TraitComponent<U>, T>
260 void addTraitComponent(T &&existing) {
261 if (static_cast<U *>(this) == nullptr) {
262 throw std::runtime_error(
263 "Cannot add TraitComponent to object that is not of the "
264 "correct type.");
265 }
266 existing.setTypedObject(static_cast<U *>(this));
267 std::shared_ptr<T> component =
268 std::make_shared<T>(std::forward<T>(existing));
269 component->object = this;
270 components.push_back(component);
271 }
272
273 /**
274 * @brief Gets the first component that matches the specified type.
275 *
276 * @tparam T The type of component to get.
277 * @return (std::shared_ptr<T>) A shared pointer to the component if found,
278 * or nullptr if not found.
279 */
280 template <typename T>
281 requires std::is_base_of_v<Component, T>
282 std::shared_ptr<T> getComponent() {
283 for (auto &component : components) {
284 std::shared_ptr<T> casted = std::dynamic_pointer_cast<T>(component);
285 if (casted != nullptr) {
286 return casted;
287 }
288 }
289 return nullptr;
290 }
291
292 /**
293 * @brief Gets the physics body associated with the object.
294 *
295 * @return A shared pointer to the physics body, or nullptr if not set.
296 */
297 std::shared_ptr<Body> body = nullptr;
298
299 protected:
300 std::vector<std::shared_ptr<Component>> components;
301};

315 * // Set up body properties...
316 * addObject(body);
317 * }
318 *
319 * void updateObjects(Window &window) override {
320 * // Update car-specific logic, e.g., movement
321 * }
322 * };
323 * ```
324 *
325 */
326class CompoundObject : public GameObject {

329 * @brief The objects that make up the compound object.

331 */

333

335
336 virtual void update(Window &window) override;
337 /**
338 * @brief Updates the objects within the compound object.

341 */
342 virtual void updateObjects(Window &window) {};
343 /**
344 * @brief Initializes the compound object.

347 virtual void init() {};
348
349 void render(float dt) override;
350 void setViewMatrix(const glm::mat4 &view) override;
