26/**
27 * @brief Behavior or property that can be attached to a \ref GameObject to
28 * extend its capabilities.
29 *
30 * \subsection example-component Example of a component
31 * ```cpp
32 * // Define a custom component by inheriting from Component
33 * class CustomComponent : public Component {
34 * public:
35 * void init() override {
36 * // Custom initialization code here
37 * }
38 * void update(float deltaTime) override {
39 * // Custom update code here
40 * }
41 * };
42 * ```
43 */
44class Component {
45 public:
46 /**
47 * @brief Initializes the component. This method is called once when the
48 * component is added to a GameObject.
49 *
50 */
51 virtual void init() {}
52 /**
53 * @brief Updates the component each frame. This method is called every
54 * frame before rendering.
55 *
56 * @param deltaTime The time elapsed since the last frame.
57 */
58 virtual void update([[maybe_unused]] float deltaTime) {}
59 /**
60 * @brief Performs changes when the GameObject's view matrix is updated.
61 *
62 * @param view The new view matrix.
63 */
64 virtual void setViewMatrix([[maybe_unused]] const glm::mat4 &view) {}
65 /**
66 * @brief Performs changes when the GameObject's projection matrix is
67 * updated.
68 *
69 * @param projection The new projection matrix.
70 */
71 virtual void
72 setProjectionMatrix([[maybe_unused]] const glm::mat4 &projection) {}
73 /**
74 * @brief Gets the window associated with the component's GameObject.
75 *
76 * @return (Window*) The window instance.
77 */
78 Window *getWindow();
79
80 /**
81 * @brief Construct a new Component object
82 *
83 */
84 Component() = default;
85
86 /**
87 * @brief Gets the GameObject associated with the component.
88 *
89 */
90 GameObject *object = nullptr;
91 /**
92 * @brief Gets the Body associated with the component.
93 *
94 */
95 Body *body = nullptr;
96};

98/**
99 * @brief Component that is specifically tied to a certain type of \ref
100 * GameObject. It cannot be added to objects that do not inherit from the
101 * specified type.
102 *
103 * @tparam T The type of GameObject the component is tied to.
104 */
105template <typename T>
106 requires std::is_base_of_v<GameObject, T>
107class TraitComponent : public Component {
108 public:
109 /**
110 * @brief Initializes the component. This method is called once when the
111 * component is added to a GameObject.
112 *
113 */
114 virtual void init() override {};
115 /**
116 * @brief Updates the component each frame. This method is called every
117 * frame before rendering.
118 *
119 * @param deltaTime The time elapsed since the last frame.
120 */
121 void update([[maybe_unused]] float deltaTime) override {
122 if (typedObject != nullptr) {
123 updateComponent(typedObject);
124 }
125 }
126 /**
127 * @brief Updates the component with a typed reference to the GameObject it
128 * is bound to.
129 *
130 * @param object A pointer to the typed GameObject.
131 */
132 virtual void updateComponent([[maybe_unused]] T *object) {}
133 /**
134 * @brief Sets the typed object reference for the component.
135 *
136 * @param obj A pointer to the typed GameObject.
137 */
138 inline void setTypedObject([[maybe_unused]] T *obj) { typedObject = obj; }
139
140 /**
141 * @brief Provides direct access to the specialized GameObject this trait
142 * decorates.
143 *
144 * @return (T*) Pointer to the typed object, or nullptr if the component
145 * has not been attached yet.
146 */
147 inline T *getObject() { return typedObject; }
148
149 private:
150 T *typedObject = nullptr;
151};

242 /**
243 * @brief Adds a component to the object.
244 *
245 * @tparam T The type of component to add.
246 * @param existing The component instance to add. \warning It must be
247 * long-lived. This means that declaring it as a class property is a good
248 * idea.
249 */
250 template <typename T>
251 requires std::is_base_of_v<Component, T>
252 void addComponent(T &&existing) {
253 std::shared_ptr<T> component =
254 std::make_shared<T>(std::forward<T>(existing));
255 component->object = this;
256 component->body = this->body.get();
257 components.push_back(component);
258 }
259
260 /**
261 * @brief Adds a trait component to the object. A trait component is a
262 * component that is tied to a specific type of GameObject.
263 *
264 * @tparam U The type of GameObject the trait component is tied to.
265 * @tparam T The type of trait component to add.
266 * @param existing The trait component instance to add. \warning It must be
267 * long-lived. This means that declaring it as a class property is a good
268 * idea.
269 */
270 template <typename U, typename T>
271 requires std::is_base_of_v<TraitComponent<U>, T>
272 void addTraitComponent(T &&existing) {
273 if (static_cast<U *>(this) == nullptr) {
274 throw std::runtime_error(
275 "Cannot add TraitComponent to object that is not of the "
276 "correct type.");
277 }
278 existing.setTypedObject(static_cast<U *>(this));
279 std::shared_ptr<T> component =
280 std::make_shared<T>(std::forward<T>(existing));
281 component->object = this;
282 components.push_back(component);
283 }
284
285 /**
286 * @brief Gets the first component that matches the specified type.
287 *
288 * @tparam T The type of component to get.
289 * @return (std::shared_ptr<T>) A shared pointer to the component if found,
290 * or nullptr if not found.
291 */
292 template <typename T>
293 requires std::is_base_of_v<Component, T>
294 std::shared_ptr<T> getComponent() {
295 for (auto &component : components) {
296 std::shared_ptr<T> casted = std::dynamic_pointer_cast<T>(component);
297 if (casted != nullptr) {
298 return casted;
299 }
300 }
301 return nullptr;
302 }
303
304 /**
305 * @brief Gets the physics body associated with the object.
306 *
307 * @return A shared pointer to the physics body, or nullptr if not set.
308 */
309 std::shared_ptr<Body> body = nullptr;
310
311 protected:
312 std::vector<std::shared_ptr<Component>> components;
313};
