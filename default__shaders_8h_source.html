222vec4 enableTextures(int type) {
223 vec4 color = vec4(0.0);
224 int count = 0;
225 for (int i = 0; i < textureCount; i++) {
226 if (textureTypes[i] == type) {
227 if (i == 0) color += texture(texture1, TexCoord);
228 else if (i == 1) color += texture(texture2, TexCoord);
229 else if (i == 2) color += texture(texture3, TexCoord);
230 else if (i == 3) color += texture(texture4, TexCoord);
231 else if (i == 4) color += texture(texture5, TexCoord);
232 else if (i == 5) color += texture(texture6, TexCoord);
233 else if (i == 6) color += texture(texture7, TexCoord);
234 else if (i == 7) color += texture(texture8, TexCoord);
235 else if (i == 8) color += texture(texture9, TexCoord);
236 else if (i == 9) color += texture(texture10, TexCoord);
237 else if (i == 10) color += texture(texture11, TexCoord);
238 else if (i == 11) color += texture(texture12, TexCoord);

282vec3 getSpecularColor() {

284 vec3 specColor = material.specular;
285 if (specTex.r != -1.0 || specTex.g != -1.0 || specTex.b != -1.0) {

287 }

302// ----- Directional Light -----
303vec3 calcDirectionalDiffuse(DirectionalLight light, vec3 norm) {
304 vec3 lightDir = normalize(-light.direction);
305 float diff = max(dot(norm, lightDir), 0.0);
306 return diff * light.diffuse;
307}

309vec3 calcDirectionalSpecular(DirectionalLight light, vec3 norm, vec3 viewDir, vec3 specColor, float shininess) {
310 vec3 lightDir = normalize(-light.direction);
311 vec3 halfwayDir = normalize(lightDir + viewDir);
312 float spec = pow(max(dot(Normal, halfwayDir), 0.0), shininess);
313 return spec * specColor * light.specular;
314}

316vec3 calcAllDirectionalLights(vec3 norm, vec3 viewDir) {
317 vec3 diffuseSum = vec3(0.0);
318 vec3 specularSum = vec3(0.0);
319 vec3 specColor = getSpecularColor();
320
321 for (int i = 0; i < directionalLightCount; i++) {
322 diffuseSum += calcDirectionalDiffuse(directionalLights[i], norm);
323 specularSum += calcDirectionalSpecular(directionalLights[i], norm, viewDir, specColor, material.shininess);
324 }
325
326 diffuseSum *= material.diffuse;
327 return diffuseSum + specularSum;
328}

337vec3 calcPointSpecular(PointLight light, vec3 norm, vec3 fragPos, vec3 viewDir, vec3 specColor, float shininess) {
338 vec3 lightDir = normalize(light.position - fragPos);
339 vec3 halfwayDir = normalize(lightDir + viewDir);

344float calcAttenuation(PointLight light, vec3 fragPos) {
345 float distance = length(light.position - fragPos);
346 return 1.0 / (light.constant + light.linear * distance + light.quadratic * distance);
347}

349vec3 calcAllPointLights(vec3 norm, vec3 fragPos, vec3 viewDir) {
350 vec3 diffuseSum = vec3(0.0);
351 vec3 specularSum = vec3(0.0);
352 vec3 specColor = getSpecularColor();
353
354 for (int i = 0; i < pointLightCount; i++) {
355 float attenuation = calcAttenuation(pointLights[i], fragPos);
356 diffuseSum += calcPointDiffuse(pointLights[i], norm, fragPos) * attenuation;
357 specularSum += calcPointSpecular(pointLights[i], norm, fragPos, viewDir, specColor, material.shininess) * attenuation;
358 }
359
360 diffuseSum *= material.diffuse;
361 return diffuseSum + specularSum;
362}

364// ----- Spot Light -----
365vec3 calcSpotDiffuse(SpotLight light, vec3 norm, vec3 fragPos) {
366 vec3 lightDir = normalize(light.position - fragPos);
367 float diff = max(dot(norm, lightDir), 0.0);
368
369 vec3 spotDirection = normalize(light.direction);
370 float theta = dot(lightDir, -spotDirection);
371
372 float intensity = smoothstep(light.outerCutOff, light.cutOff, theta);
373
374 return diff * light.diffuse * intensity;
375}

378vec3 calcSpotSpecular(SpotLight light, vec3 norm, vec3 fragPos, vec3 viewDir, vec3 specColor, float shininess) {
379 vec3 lightDir = normalize(light.position - fragPos);
380 vec3 halfwayDir = normalize(lightDir + viewDir);
381 float spec = pow(max(dot(Normal, halfwayDir), 0.0), shininess);
382
383 vec3 spotDirection = normalize(light.direction);
384 float theta = dot(lightDir, -spotDirection);
385
386 float intensity = smoothstep(light.outerCutOff, light.cutOff, theta);
387
388 return spec * specColor * light.specular * intensity;
389}

392float calcSpotAttenuation(SpotLight light, vec3 fragPos) {
393 float distance = length(light.position - fragPos);
394 return 1.0 / (1.0 + 0.09 * distance + 0.032 * distance);
395}

397vec3 calcAllSpotLights(vec3 norm, vec3 fragPos, vec3 viewDir) {
398 vec3 diffuseSum = vec3(0.0);
399 vec3 specularSum = vec3(0.0);
400 vec3 specColor = getSpecularColor();
401
402 for (int i = 0; i < spotlightCount; i++) {
403 float attenuation = calcSpotAttenuation(spotlights[i], fragPos);
404 diffuseSum += calcSpotDiffuse(spotlights[i], norm, fragPos) * attenuation;
405 specularSum += calcSpotSpecular(spotlights[i], norm, fragPos, viewDir, specColor, material.shininess) * attenuation;
406 }
407
408 diffuseSum *= material.diffuse;
409 return diffuseSum + specularSum;

412// ----- Shadow Calculations -----
413float calculateShadow(ShadowParameters shadowParam, vec4 fragPosLightSpace) {
414 vec3 projCoords = fragPosLightSpace.xyz / fragPosLightSpace.w;
415 projCoords = projCoords * 0.5 + 0.5;
416
417 if (projCoords.x < 0.0 || projCoords.x > 1.0 ||
418 projCoords.y < 0.0 || projCoords.y > 1.0 ||
419 projCoords.z > 1.0) {
420 return 0.0;
421 }
422
423 float currentDepth = projCoords.z;
424
425 vec3 lightDir = normalize(-directionalLights[0].direction);
426 vec3 normal = normalize(Normal);
427 float biasValue = shadowParam.bias;
428 float bias = max(biasValue * (1.0 - dot(normal, lightDir)), biasValue);
429
430 float shadow = 0.0;
431 vec2 texelSize = 1.0 / getTextureDimensions(shadowParam.textureIndex);
432
433 float distance = length(cameraPosition - FragPos);
434 int kernelSize = int(mix(1.0, 3.0, clamp(distance / 100.0, 0.0, 1.0)));
435
436 int sampleCount = 0;
437 for(int x = -kernelSize; x <= kernelSize; ++x) {
438 for(int y = -kernelSize; y <= kernelSize; ++y) {
439 float pcfDepth = sampleTextureAt(shadowParam.textureIndex,
440 projCoords.xy + vec2(x, y) * texelSize).r;
441 shadow += currentDepth - bias > pcfDepth ? 1.0 : 0.0;
442 sampleCount++;
443 }
444 }
445 shadow /= float(sampleCount);
446
447 return shadow;
448}

450float calculateShadowRaw(ShadowParameters shadowParam, vec4 fragPosLightSpace) {
451 vec3 projCoords = fragPosLightSpace.xyz / fragPosLightSpace.w;
452 projCoords = projCoords * 0.5 + 0.5;
453
454 if (projCoords.x < 0.0 || projCoords.x > 1.0 ||
455 projCoords.y < 0.0 || projCoords.y > 1.0 ||
456 projCoords.z < 0.0 || projCoords.z > 1.0) {
457 return 0.0;
458 }
459
460 float currentDepth = projCoords.z;
461 float closestDepth = sampleTextureAt(shadowParam.textureIndex, projCoords.xy).r;
462
463 return currentDepth > closestDepth ? 1.0 : 0.0;
464}

466float calculateAllShadows() {
467 float totalShadow = 0.0;
468 for (int i = 0; i < shadowParamCount; i++) {
469 vec4 fragPosLightSpace = shadowParams[i].lightProjection * shadowParams[i].lightView * vec4(FragPos, 1.0);
470 float shadow = calculateShadow(shadowParams[i], fragPosLightSpace);
471 totalShadow = max(totalShadow, shadow);
472 }
473 return totalShadow;
474}

476// ----- Main -----
477void main() {
478 vec4 baseColor;
479
480 if (useTexture && !useColor)
481 baseColor = enableTextures(TEXTURE_COLOR);
482 else if (useTexture && useColor)
483 baseColor = enableTextures(TEXTURE_COLOR) * outColor;
484 else if (!useTexture && useColor)
485 baseColor = outColor;
486 else
487 baseColor = vec4(1.0);
488
489 vec3 norm = normalize(Normal);
490 vec3 viewDir = normalize(cameraPosition - FragPos);
491
492 vec3 ambient = ambientLight.color.rgb * ambientLight.intensity * material.ambient;
493 vec3 directionalLights = calcAllDirectionalLights(norm, viewDir);
494 vec3 pointLights = calcAllPointLights(norm, FragPos, viewDir);
495 vec3 spotLightsContrib = calcAllSpotLights(norm, FragPos, viewDir);
496
497 vec3 lightContribution = directionalLights + pointLights + spotLightsContrib;
498
499 float shadow = calculateAllShadows();
500
501 vec3 finalColor = (ambient + (1.0 - shadow) * lightContribution) * baseColor.rgb;
502
503 FragColor = vec4(finalColor, baseColor.a);
504 FragColor = getEnvironmentReflected(FragColor);
505
506 if (FragColor.a < 0.1)
507 discard;
508}
509
510)";
