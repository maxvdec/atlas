24/**
25 * @brief Dynamic water surface that captures reflections/refractions and
26 * renders Gerstner-style waves.
27 *
28 * \subsection fluid-example Example
29 * ```cpp
30 * // Create a water surface with specific extent and color
31 * Fluid waterSurface;
32 * waterSurface.create(Size2d(100.0f, 100.0f), Color(0.1, 0.3, 0.6, 0.8));
33 * waterSurface.setPosition(Position3d(0.0f, -2.0f, 0.0f));
34 *
35 * // Configure wave animation
36 * waterSurface.setWaveVelocity(0.5f); // Wave movement speed
37 *
38 * // Load normal and movement textures for surface detail (via Workspace)
39 * Workspace::get().createResource("textures/water_normal.png", "WaterNormal",
40 * ResourceType::Image);
41 * Workspace::get().createResource("textures/water_flow.png", "WaterFlow",
42 * ResourceType::Image);
43 * waterSurface.normalTexture =
44 * Texture::fromResourceName("WaterNormal", TextureType::Normal);
45 * waterSurface.movementTexture =
46 * Texture::fromResourceName("WaterFlow", TextureType::Color);
47 *
48 * // Initialize and add to scene
49 * waterSurface.initialize();
50 * scene.addObject(&waterSurface);
51 *
52 * // The fluid automatically captures reflections and refractions
53 * // during the render pass
54 * ```
55 *
56 */
