29 * @brief Spatial layout and atlas packing configuration for DDGI probes.

31struct ProbeSpace {
32 /** @brief World-space origin of probe index (0,0,0). */
33 Position3d originWorldSpace;
34 /** @brief Distance between neighboring probes on each axis. */
35 Position3d spacing;
36 /** @brief Probe counts along X/Y/Z axes. */
37 Vector3 probeCount;
38 /** @brief Debug tint used by probe visualization tools. */
39 Color debugColor;
40
41 /** @brief Border texels used around each probe tile to reduce seams. */
42 int textureBorderSize = 1;
43 /** @brief Inner texel resolution per probe tile (without border). */
44 int probeResolution = 6;
45 /** @brief Number of probe tiles packed per atlas row. */
46 int probesPerRow = 16;
47
48 /** @brief Returns the total number of probes in the grid. */
49 int totalProbes() const {
50 return static_cast<int>(probeCount.x * probeCount.y * probeCount.z);
51 }
52
53 /** @brief Returns per-tile resolution including border texels. */
54 int tileResolution() const {
55 return probeResolution + (2 * textureBorderSize);
56 }
57
58 /** @brief Returns the number of rows required by the probe atlas. */
59 int atlasRows() const {
60 return (totalProbes() + probesPerRow - 1) / probesPerRow;
61 }
62
63 /** @brief Returns the packed atlas width in pixels. */
64 int atlasWidth() const { return probesPerRow * tileResolution(); }
65
66 /** @brief Returns the packed atlas height in pixels. */
67 int atlasHeight() const { return atlasRows() * tileResolution(); }
68};
