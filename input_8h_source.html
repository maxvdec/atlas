214/**
215 * @brief Abstract class that declares some object that can recieve input and
216 * react to the changes in it.
217 * \subsection interactive-example Example
218 * ```cpp
219 * class PlayerController : public Interactive {
220 * public:
221 * void onKeyPress(Key key, float deltaTime) override {
222 * if (key == Key::W) {
223 * // Move forward
224 * }
225 * }
226 * void onMouseMove(MousePacket data, float deltaTime) override {
227 * // Look around based on mouse movement
228 * }
229 * };
230 * ```
231 *
232 */
233class Interactive {
234 public:
235 /**
236 * @brief Function that is called automatically when a key is pressed.
237 *
238 * @param key The key that was pressed.
239 * @param deltaTime The time elapsed since the last frame.
240 */
241 virtual void onKeyPress(Key key, float deltaTime) {};
242 /**
243 * @brief Function that is called automatically when a key is released.
244 *
245 * @param key The key that was released.
246 * @param deltaTime The time elapsed since the last frame.
247 */
248 virtual void onKeyRelease(Key key, float deltaTime) {};
249 /**
250 * @brief Function that is called automatically when the mouse is moved.
251 *
252 * @param data The mouse movement data. See \ref MousePacket.
253 * @param deltaTime The time elapsed since the last frame.
254 */
255 virtual void onMouseMove(MousePacket data, float deltaTime) {};
256 /**
257 * @brief Function that is called automatically when a mouse button is
258 * pressed.
259 *
260 * @param button The mouse button that was pressed.
261 * @param deltaTime The time elapsed since the last frame.
262 */
263 virtual void onMouseButtonPress(MouseButton button, float deltaTime) {};
264 /**
265 * @brief Function that is called automatically when the mouse is scrolled.
266 *
267 * @param data The mouse scroll data. See \ref MouseScrollPacket.
268 * @param deltaTime The time elapsed since the last frame.
269 */
270 virtual void onMouseScroll(MouseScrollPacket data, float deltaTime) {};
271 /**
272 * @brief Function that is called automatically at each frame.
273 *
274 * @param deltaTime The time elapsed since the last frame.
275 */
276 virtual void atEachFrame(float deltaTime) {};
277
278 /**
279 * @brief The last mouse data received. Updated automatically by the \ref
280 * Window class.
281 *
282 */
283 MousePacket lastMouseData = MousePacket();
284};
