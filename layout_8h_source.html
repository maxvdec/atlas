19/**
20 * @brief Alignment options used when positioning children inside layouts.
21 */
22enum class ElementAlignment {
23 /** @brief Align children to the start edge of the available axis. */
24 Top,
25 /** @brief Center children along the available axis. */
26 Center,
27 /** @brief Align children to the end edge of the available axis. */
28 Bottom,
29};

31/**

35enum class LayoutAnchor {

37 TopLeft,
38 /** @brief Interpret position as the container's top edge center. */

41 TopRight,
42 /** @brief Interpret position as the container's center-left edge. */
43 CenterLeft,
44 /** @brief Interpret position as the container's geometric center. */

46 /** @brief Interpret position as the container's center-right edge. */

69/**
70 * @brief Vertical layout that stacks children from top to bottom.
71 *
72 * \subsection graphite-column-example Example
73 * ```cpp
74 * Text title("Settings", Font::getFont("Inter"), Color::white());
75 * Button save(Font::getFont("Inter"), "Save");
76 *
77 * Column panel({&title, &save}, 12.0f, {.width = 16.0f, .height = 16.0f},
78 * {.x = 24.0f, .y = 24.0f});
79 * panel.alignment = ElementAlignment::Center;
80 * panel.style().normal().background(Color(0.1f, 0.11f, 0.15f, 0.94f));
81 * ```
82 */
83class Column : public UIObject {
84 public:
85 /** @brief Constructs an empty column at the given position. */
86 Column(Position2d pos = {.x = 0, .y = 0}) : position(pos) {
87 recalculatePositions();
88 }
89 /** @brief Constructs a column with children, spacing, padding, and position. */
90 Column(const std::vector<UIObject *> &children, float spacing = 0.0f,
91 Size2d padding = {.width = 0.0f, .height = 0.0f},
92 Position2d pos = {.x = 0, .y = 0})
93 : spacing(spacing), padding(padding), children(children),
94 position(pos) {
95 recalculatePositions();
96 }
97 /** @brief Renders the column background and all children. */
98 void render(float dt, std::shared_ptr<opal::CommandBuffer> commandBuffer,
99 bool updatePipeline) override;
100
101 /** @brief Spacing inserted between consecutive children. */
102 float spacing = 0.0f;
103 /** @brief Minimum size enforced for the layout container. */
104 Size2d maxSize{.width = 0.0f, .height = 0.0f};
105 /** @brief Padding applied around the children. */
106 Size2d padding{.width = 0.0f, .height = 0.0f};
107 /** @brief Child elements owned externally by the caller. */
108 std::vector<UIObject *> children;
109 /** @brief Screen-space anchor position of the layout container. */
110 Position2d position;
111 /** @brief Horizontal alignment applied to each child within the column. */
112 ElementAlignment alignment = ElementAlignment::Top;
113 /** @brief Anchor mode used to interpret position. */
114 LayoutAnchor anchor = LayoutAnchor::TopLeft;
115
116 /** @brief Propagates the view matrix to all children. */
117 void setViewMatrix(const glm::mat4 &view) override;
118 /** @brief Propagates the projection matrix to all children. */
119 void setProjectionMatrix(const glm::mat4 &projection) override;
120
121 /** @brief Appends a child to the layout. */
122 void addChild(UIObject *child);
123 /** @brief Replaces the child list for the layout. */
124 void setChildren(const std::vector<UIObject *> &newChildren);
125
126 /** @brief Computes the layout size including padding and child spacing. */
127 Size2d getSize() const override {
128 graphite::UIStyle fallbackStyle;
129 fallbackStyle.normal().padding(padding);
130 const graphite::UIResolvedStyle style = graphite::resolveStyle(
131 fallbackStyle, &graphite::Theme::current().column,
132 usesLocalStyle ? &localStyle : nullptr);
133 const Size2d effectivePadding = style.padding;
134 float width = 0.0f;
135 float totalHeight = effectivePadding.height * 2.0f;
136 bool hasChild = false;
137 for (const auto *child : children) {
138 if (child == nullptr) {
139 continue;
140 }
141 Size2d childSize = child->getSize();
142 width = std::max(width, childSize.width);
143 totalHeight += childSize.height;
144 if (hasChild) {
145 totalHeight += spacing;
146 }
147 hasChild = true;
148 }
149 return Size2d{
150 .width =
151 std::max(width + (effectivePadding.width * 2.0f), maxSize.width),
152 .height = std::max(totalHeight, maxSize.height),
153 };
154 }
155
156 /** @brief Returns the layout screen position. */
157 Position2d getScreenPosition() const override { return position; }
158 /** @brief Sets the layout screen position and recomputes child placement. */
159 void setScreenPosition(const Position2d &newPosition) override {
160 position = newPosition;
161 recalculatePositions();
162 }
163
164 /** @brief Returns a mutable local style override for this layout. */
165 graphite::UIStyle &style() {
166 usesLocalStyle = true;
167 return localStyle;
168 }
169
170 /** @brief Replaces the local style override for this layout. */
171 Column &setStyle(const graphite::UIStyle &newStyle) {
172 localStyle = newStyle;
173 usesLocalStyle = true;
174 recalculatePositions();
175 return *this;
176 }
177
178 private:
179 void recalculatePositions();
180 graphite::BoxRendererData boxRenderer;
181 graphite::UIStyle localStyle;
182 bool usesLocalStyle = false;
183};
