40/**
41 * @brief Structure representing the material properties of an object. Based on
42 * Phong reflection model.
43 *
44 */
45struct Material {
46 /**
47 * @brief The ambient color of the material. This is the color that the
48 * material appears to be under ambient lighting.
49 *
50 */
51 Color ambient = Color::white();
52 /**
53 * @brief The diffuse color of the material. This is the color that the
54 * material appears to be under diffuse lighting.
55 *
56 */
57 Color diffuse = Color::white();
58 /**
59 * @brief The specular color of the material. This is the color that the
60 * material appears to be under specular lighting.
61 *
62 */
63 Color specular = Color::white();
64 /**
65 * @brief The shininess of the material. This determines the size and
66 * intensity of specular highlights. \warning The values must be greater
67 * than 0.
68 */
69 float shininess = 32.0f;
70
71 float reflectivity = 0.f; // 0.0 = no reflection, 1.0 = full reflection
72};

74/**
75 * @brief Structure representing a single vertex in 3D space, including
76 * position, color, texture coordinates, and normal vector. \warning This is
77 * meant for internal use only and should not be used directly because it's
78 * difficult to read and maintain.
79 * \subsection core-vertex-example Example
80 * ```cpp
81 * // Create a vertex at position (1, 2, 3) with red color
82 * CoreVertex vertex({1.0, 2.0, 3.0}, {1.0, 0.0, 0.0, 1.0}, {0.5, 0.5}, {0.0,
83 * 0.0, 1.0});
84 * ```
85 */
86struct CoreVertex {
87 /**
88 * @brief The position of the vertex in 3D space.
89 *
90 */
91 Position3d position;
92 /**
93 * @brief The color of the vertex.
94 *
95 */
96 Color color = {1.0, 1.0, 1.0, 1.0};
97 /**
98 * @brief The texture coordinates of the vertex.
99 *
100 */
101 TextureCoordinate textureCoordinate = {0.0, 0.0};
102 /**
103 * @brief The normal vector of the vertex, used for lighting calculations.
104 *
105 */
106 Normal3d normal = {0.0, 0.0, 0.0};
107
108 Normal3d tangent = {0.0, 0.0, 0.0};
109 Normal3d bitangent = {0.0, 0.0, 0.0};

111 /**
112 * @brief Function that constructs a new CoreVertex object.

114 * @param pos The position of the vertex in 3D space.
