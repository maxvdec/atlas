64/**
65 * @brief Structure representing a single vertex in 3D space, including
66 * position, color, texture coordinates, and normal vector. \warning This is
67 * meant for internal use only and should not be used directly because it's
68 * difficult to read and maintain.
69 * \subsection core-vertex-example Example
70 * ```cpp
71 * // Create a vertex at position (1, 2, 3) with red color
72 * CoreVertex vertex({1.0, 2.0, 3.0}, {1.0, 0.0, 0.0, 1.0}, {0.5, 0.5}, {0.0,
73 * 0.0, 1.0});
74 * ```
75 */
76struct CoreVertex {
77 /**
78 * @brief The position of the vertex in 3D space.
79 *
80 */
81 Position3d position;
82 /**
83 * @brief The color of the vertex.
84 *
85 */
86 Color color = {1.0, 1.0, 1.0, 1.0};
87 /**
88 * @brief The texture coordinates of the vertex.
89 *
90 */
91 TextureCoordinate textureCoordinate = {0.0, 0.0};
92 /**
93 * @brief The normal vector of the vertex, used for lighting calculations.
94 *
95 */
96 Normal3d normal = {0.0, 0.0, 0.0};
97
98 /**
99 * @brief The tangent vector of the vertex, used for normal mapping and
100 * parallax calculations.
101 *
102 */
103 Normal3d tangent = {0.0, 0.0, 0.0};
104 /**
105 * @brief The bitangent vector of the vertex, used for normal mapping and
106 * parallax calculations.
107 *
108 */
109 Normal3d bitangent = {0.0, 0.0, 0.0};
110
111 /**
112 * @brief Function that constructs a new CoreVertex object.
113 *
114 * @param pos The position of the vertex in 3D space.
115 * @param col The color of the vertex.
116 * @param tex The texture coordinates of the vertex.
117 * @param n The normal vector of the vertex.
118 */
119 CoreVertex(Position3d pos = {0.0, 0.0, 0.0},
120 Color col = {1.0, 1.0, 1.0, 1.0},
121 TextureCoordinate tex = {0.0, 0.0}, Normal3d n = {0.0, 0.0, 0.0},
122 Normal3d t = {0.0, 0.0, 0.0}, Normal3d b = {0.0, 0.0, 0.0})
123 : position(pos), color(col), textureCoordinate(tex), normal(n),
124 tangent(t), bitangent(b) {}
125
126 /**
127 * @brief Gets the layout descriptors for the CoreVertex structure.
128 *
129 * @return (std::vector<LayoutDescriptor>) The layout descriptors.
130 */
131 static std::vector<LayoutDescriptor> getLayoutDescriptors();
132};

147/**
148 * @brief Structure representing a single instance of an object for instanced
149 * rendering. Each instance has its own position, rotation, and scale.
150 *
151 */
152struct Instance {
153 /**
154 * @brief The position of this instance in 3D space.
155 *
156 */
157 Position3d position = {0.0, 0.0, 0.0};
158 /**
159 * @brief The rotation of this instance in 3D space.
160 *
161 */
162 Rotation3d rotation = {0.0, 0.0, 0.0};
163 /**
164 * @brief The scale of this instance in 3D space.
165 *
166 */
167 Scale3d scale = {1.0, 1.0, 1.0};
168
169 private:
170 glm::mat4 model = glm::mat4(1.0f);
171
172 public:
173 /**
174 * @brief Updates the model matrix based on the instance's position,
175 * rotation, and scale.
176 *
177 */
178 void updateModelMatrix();
179 /**
180 * @brief Gets the current model matrix for this instance.
181 *
182 * @return (glm::mat4) The model matrix.
183 */
184 glm::mat4 getModelMatrix() const { return model; }
185 /**
186 * @brief Moves the instance by a delta position.
187 *
188 * @param deltaPosition The amount to move by.
189 */
190 void move(const Position3d &deltaPosition);
191 /**
192 * @brief Sets the position of the instance.
193 *
194 * @param newPosition The new position to set.
195 */
196 void setPosition(const Position3d &newPosition);
197 /**
198 * @brief Sets the rotation of the instance.
199 *
200 * @param newRotation The new rotation to set.
201 */
202 void setRotation(const Rotation3d &newRotation);
203 /**
204 * @brief Rotates the instance by a delta rotation.
205 *
206 * @param deltaRotation The amount to rotate by.
207 */
208 void rotate(const Rotation3d &deltaRotation);
209 /**
210 * @brief Sets the scale of the instance.
211 *
212 * @param newScale The new scale to set.
213 */
214 void setScale(const Scale3d &newScale);
215 /**
216 * @brief Scales the instance by a delta scale factor.
217 *
218 * @param deltaScale The scale factor to multiply by.
219 */
220 void scaleBy(const Scale3d &deltaScale);
221
222 /**
223 * @brief Compares two instances for equality.
224 *
225 * @param other The other instance to compare with.
226 * @return (bool) True if instances are equal, false otherwise.
227 */
228 bool operator==(const Instance &other) const {
229 return position == other.position && rotation == other.rotation &&
230 scale == other.scale;
231 }

235 * @brief Represents a 3D object in the scene, including its geometry, material

243 * CoreObject cube = createBox({1.0, 1.0, 1.0}, Color::red());

245 * cube.setPosition({0.0, 0.5, 0.0});
246 * // Attach a texture to the cube
247 * Texture brickTexture("path/to/brick_texture.png");
248 * cube.attachTexture(brickTexture);
249 * // Create and attach a shader program to the cube
250 * VertexShader vertexShader("path/to/vertex_shader.glsl");
251 * FragmentShader fragmentShader("path/to/fragment_shader.glsl");
252 * cube.createAndAttachProgram(vertexShader, fragmentShader);
253 * // Set the material properties of the cube
254 * Material cubeMaterial;
255 * cubeMaterial.ambient = Color(0.2, 0.2, 0.2, 1.0);
256 * cubeMaterial.diffuse = Color(0.8, 0.0, 0.0, 1.0);
257 * cubeMaterial.specular = Color(1.0, 1.0, 1.0, 1.0);

262 * cubeBody.type = BodyType::Dynamic;
263 * cubeBody.mass = 1.0f;
264 * cubeBody.friction = 0.5f;

266 * // Add the cube to the scene
