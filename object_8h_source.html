40 *
41 */
42struct Material {
43 /**
44 * @brief The ambient color of the material. This is the color that the
45 * material appears to be under ambient lighting.
46 *
47 */
48 Color ambient = Color::white();
49 /**
50 * @brief The diffuse color of the material. This is the color that the
51 * material appears to be under diffuse lighting.
52 *
53 */
54 Color diffuse = Color::white();

56 * @brief The specular color of the material. This is the color that the
57 * material appears to be under specular lighting.
58 *
59 */

62 * @brief The shininess of the material. This determines the size and
