20/**
21 * @file atlas/physics.h
22 * @brief High-level physics components and query API for Atlas.
23 *
24 * This header provides engine-facing physics components (e.g. `Rigidbody`,
25 * `Vehicle`, joints) and query result containers (raycasts, overlaps, sweeps).
26 *
27 * \note This is an alpha API and may change.
28 *
29 * \subsection physics-example Example
30 * ```cpp
31 * // Attach physics to a GameObject via components
32 * auto rb = std::make_shared<Rigidbody>();
33 * rb->setMotionType(MotionType::Dynamic);
34 * rb->setMass(5.0f);
35 * rb->addBoxCollider({0.5f, 0.5f, 0.5f});
36 * rb->setFriction(0.8f);
37 *
38 * // Optional tagging for filtered queries
39 * rb->addTag("Player");
40 *
41 * // Trigger a raycast query (results are delivered via onQueryRecieve)
42 * rb->raycast({0.0f, -1.0f, 0.0f}, 100.0f);
43 *
44 * // For sensors, you can emit signals on contact
45 * auto sensor = std::make_shared<Sensor>();
46 * sensor->addSphereCollider(1.0f);
47 * sensor->setSignal("EnteredTrigger");
48 * ```
49 */
50

53/**
54 * @brief Single hit returned by a raycast query.
55 */
56struct RaycastHit {
57 /** @brief World-space impact position. */
58 Position3d position = {0.0f, 0.0f, 0.0f};
59 /** @brief Surface normal at the impact point. */
60 Normal3d normal = {0.0f, 0.0f, 0.0f};
61 /** @brief Distance from ray origin to the hit point. */
62 float distance = 0.0;
63 /** @brief Atlas-side object that owns the rigidbody, when known. */
64 GameObject *object = nullptr;
65 /** @brief Underlying Bezel rigidbody pointer, when known. */
66 bezel::Rigidbody *rigidbody = nullptr;
67 /** @brief Whether the query produced a valid hit. */
68 bool didHit = false;
69};
70
71/**
72 * @brief Aggregated results for raycast queries.
73 */
74struct RaycastResult {
75 /** @brief All hits (for `RaycastAll` variants). */
76 std::vector<RaycastHit> hits;
77 /** @brief Primary hit (for `Raycast` variants). */
78 RaycastHit hit;
79 /** @brief Convenience distance for the nearest impact. */
80 float closestDistance = 0.0f;
81};
82
83/**
84 * @brief Single hit returned by an overlap query.
85 */
86struct OverlapHit {
87 /** @brief World-space contact point. */
88 Position3d contactPoint = {0.0f, 0.0f, 0.0f};
89 /** @brief Axis along which penetration occurs. */
90 Point3d penetrationAxis = {0.0f, 0.0f, 0.0f};
91 /** @brief Signed penetration depth (implementation-defined sign). */
92 float penetrationDepth = 0.0f;
93 /** @brief Atlas-side owner object, when known. */
94 GameObject *object = nullptr;
95 /** @brief Underlying Bezel rigidbody pointer, when known. */
96 bezel::Rigidbody *rigidbody = nullptr;
97};

99/**
100 * @brief Aggregated results for overlap queries.
101 */
102struct OverlapResult {
103 /** @brief All overlap hits. */
104 std::vector<OverlapHit> hits;
105 /** @brief Whether any overlap occurred. */
106 bool hitAny = false;
107};

109/**
110 * @brief Single hit returned by a sweep query.
111 */
112struct SweepHit {
113 /** @brief World-space impact position. */
114 Position3d position = {0.0f, 0.0f, 0.0f};
115 /** @brief Surface normal at the impact point. */
116 Normal3d normal = {0.0f, 0.0f, 0.0f};
117 /** @brief Distance traveled before impact. */
118 float distance = 0.0f;
119 /** @brief Impact fraction along the sweep in [0, 1]. */
120 float percentage = 0.0f;
121 /** @brief Atlas-side owner object, when known. */
122 GameObject *object = nullptr;
123 /** @brief Underlying Bezel rigidbody pointer, when known. */
124 bezel::Rigidbody *rigidbody = nullptr;
125};
126
127/**
128 * @brief Aggregated results for sweep (movement prediction) queries.
129 */
130struct SweepResult {
131 /** @brief All sweep hits (for "All" variants). */
132 std::vector<SweepHit> hits;
133 /** @brief Closest hit for convenience. */
134 SweepHit closest;
135 /** @brief Whether any hit occurred. */
136 bool hitAny = false;
137 /** @brief End position of the sweep when no hit blocks the movement. */
138 Position3d endPosition = {0.0f, 0.0f, 0.0f};
139};
