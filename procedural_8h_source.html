177/**
178 * @brief Low-frequency noise generator used for rolling hills and plains.
179 */
180class HillGenerator : public TerrainGenerator {
181 private:
182 float scale;
183 float amplitude;
184
185 public:
186 /**
187 * @brief Configures the hill generator parameters.
188 *
189 * @param scale Noise scale controlling hill size.
190 * @param amplitude Maximum height contribution.
191 */
192 HillGenerator(float scale = 0.01f, float amplitude = 10.0f)
193 : scale(scale), amplitude(amplitude) {}
194
195 /**
196 * @brief Samples Perlin noise and remaps it to a gentle hill profile.
197 */
198 float generateHeight(float x, float y) override {
199 float noise = Noise::perlin(x / scale, y / scale);
200 return (noise + 1.0f) * 0.5f * amplitude / 10.0f;
201 }
202};

276/**
277 * @brief Aggregates multiple terrain generators and sums their contributions.
278 */
279class CompoundGenerator : public TerrainGenerator {
280 private:
281 std::vector<std::shared_ptr<TerrainGenerator>> generators;
282
283 public:
284 /**
285 * @brief Adds another generator to the compound stack.
286 *
287 * @tparam T Type of generator to store.
288 * @param gen Instance that will be copied into the stack.
289 */
290 template <typename T>
291 requires std::is_base_of<TerrainGenerator, T>::value
292 void addGenerator(T gen) {
293 generators.push_back(std::make_shared<T>(std::move(gen)));
294 }
295
296 /**
297 * @brief Sums the height contributions of all registered generators.
298 */
299 float generateHeight(float x, float y) override {
300 float height = 0.0f;
301 for (auto &g : generators) {
302 height += g->generateHeight(x, y);
303 }
304 return height;
305 }
306};
