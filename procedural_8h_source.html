196/**
197 * @brief Low-frequency noise generator used for rolling hills and plains.
198 *
199 * \subsection terraingenerator-example Example
200 * ```cpp
201 * // Use individual generators
202 * HillGenerator hills(0.02f, 15.0f);
203 * MountainGenerator mountains(8.0f, 120.0f, 6, 0.6f);
204 *
205 * // Or combine multiple generators
206 * CompoundGenerator combined;
207 * combined.addGenerator(PlainGenerator(0.03f, 3.0f));
208 * combined.addGenerator(MountainGenerator(10.0f, 80.0f));
209 * combined.addGenerator(HillGenerator(0.01f, 20.0f));
210 *
211 * // Apply to terrain
212 * Terrain terrain;
213 * combined.applyTo(terrain);
214 *
215 * // Or sample directly
216 * float heightValue = combined.generateHeight(100.0f, 50.0f);
217 * ```
218 */
219class HillGenerator : public TerrainGenerator {
220 private:
221 float scale;
222 float amplitude;
223
224 public:
225 /**
226 * @brief Configures the hill generator parameters.
227 *
228 * @param scale Noise scale controlling hill size.
229 * @param amplitude Maximum height contribution.
230 */
231 HillGenerator(float scale = 0.01f, float amplitude = 10.0f)
232 : scale(scale), amplitude(amplitude) {}
233
234 /**
235 * @brief Samples Perlin noise and remaps it to a gentle hill profile.
236 */
237 float generateHeight(float x, float y) override {
238 float noise = Noise::perlin(x / scale, y / scale);
239 return (noise + 1.0f) * 0.5f * amplitude / 10.0f;
240 }
241};

313/**
314 * @brief Aggregates multiple terrain generators and sums their contributions.
315 */
316class CompoundGenerator : public TerrainGenerator {
317 private:
318 std::vector<std::shared_ptr<TerrainGenerator>> generators;
319
320 public:
321 /**
322 * @brief Adds another generator to the compound stack.
323 *
324 * @tparam T Type of generator to store.
325 * @param gen Instance that will be copied into the stack.
326 */
327 template <typename T>
328 requires std::is_base_of<TerrainGenerator, T>::value
329 void addGenerator(T gen) {
330 generators.push_back(std::make_shared<T>(std::move(gen)));
331 }
332
333 /**
334 * @brief Sums the height contributions of all registered generators.
335 */
336 float generateHeight(float x, float y) override {
337 float height = 0.0f;
338 for (auto &g : generators) {
339 height += g->generateHeight(x, y);
340 }
341 return height;
342 }
343};
