24/**
25 * @brief An abstract base class representing any object that can be rendered in
26 * a Window. Contains virtual methods for rendering, initialization, updating,
27 * and setting view/projection matrices.
28 *
29 * \subsection renderable-example Example
30 * ```cpp
31 * // Create a custom renderable object
32 * class CustomRenderable : public Renderable {
33 * private:
34 * Position3d position;
35 * glm::mat4 viewMatrix, projectionMatrix;
36 * std::shared_ptr<opal::Pipeline> pipeline;
37 * std::shared_ptr<opal::Buffer> vertexBuffer;
38 *
39 * public:
40 * void initialize() override {
41 * // Set up buffers, load shaders, create pipeline
42 * opal::Device& device = opal::Device::get();
43 * pipeline = device.createPipeline(...);
44 * vertexBuffer = device.createBuffer(...);
45 * }
46 *
47 * void render(float dt, std::shared_ptr<opal::CommandBuffer> commandBuffer,
48 * bool updatePipeline) override {
49 * // Bind pipeline and draw
50 * commandBuffer->bindPipeline(pipeline);
51 * commandBuffer->bindVertexBuffer(vertexBuffer);
52 * commandBuffer->draw(vertexCount, 1, 0, 0);
53 * }
54 *
55 * void update(Window& window) override {
56 * // Update logic (movement, animation, etc.)
57 * position += Position3d(0.01f, 0.0f, 0.0f);
58 * }
59 *
60 * void setViewMatrix(const glm::mat4& view) override {
61 * viewMatrix = view;
62 * }
63 *
64 * void setProjectionMatrix(const glm::mat4& projection) override {
65 * projectionMatrix = projection;
66 * }
67 *
68 * Position3d getPosition() const override { return position; }
69 * bool canCastShadows() const override { return true; }
70 * };
71 * ```
72 *
73 */
74class Renderable {
75 public:
76 /**
77 * @brief Function to render the object. Must be implemented by derived
78 * classes.
79 *
80 * @param dt Delta time since the last frame, useful for animations.
81 * @param updatePipeline When true, the renderable should rebuild or fetch
82 * the graphics pipeline to reflect the window's current state before
83 * drawing. When false, the previously prepared pipeline may be reused.
84 */
85 virtual void render(float dt,
86 std::shared_ptr<opal::CommandBuffer> commandBuffer,
87 bool updatePipeline = false) = 0;
88 /**
89 * @brief Function to initialize the object. Can be overridden by derived.
90 *
91 */
92 virtual void initialize() {};
93
94 virtual unsigned int getId() { return 0; };
95 /**
96 * @brief Function to update the object each frame.
97 * \warning It runs before the rendering phase and it should only contain
98 * logic updates.
99 *
100 * @param window The window where the object is being rendered.
101 */
102 virtual void update([[maybe_unused]] Window &window) {};
103 /**
104 * @brief Function to set the view matrix for the object. Called from \ref
105 * Window for internal purposes.
106 *
107 * @param view The view matrix to set.
108 */
109 virtual void setViewMatrix([[maybe_unused]] const glm::mat4 &view) {};
110 /**
111 * @brief Function to set the projection matrix for the object. Called from
112 * \ref Window for internal purposes.
113 *
114 * @param projection The projection matrix to set.
115 */
116 virtual void
117 setProjectionMatrix([[maybe_unused]] const glm::mat4 &projection) {};
118 virtual std::optional<std::shared_ptr<opal::Pipeline>> getPipeline() {
119 return std::nullopt;
120 };
121
122 virtual void
123 setPipeline([[maybe_unused]] std::shared_ptr<opal::Pipeline> &pipeline) {};
124
125 /**
126 * @brief Returns the currently bound shader program, if any.
127 */
128 virtual std::optional<ShaderProgram> getShaderProgram() {
129 return std::nullopt;
130 };
131
132 /**
133 * @brief Replaces the shader program bound to the renderable.
134 */
135 virtual void
136 setShader([[maybe_unused]] const ShaderProgram &shaderProgram) {};
137 /**
138 * @brief Function to get the position of the object in 3D space.
139 *
140 * @return The position of the object as a Position3d struct.
141 */
142 virtual Position3d getPosition() const { return {0.0f, 0.0f, 0.0f}; };
143 /**
144 * @brief Function to get the vertices of the object in 3D space.
145 *
146 * @return The vertices of the object as a vector of CoreVertex structs.
147 */
148 virtual std::vector<CoreVertex> getVertices() const { return {}; };
149 /**
150 * @brief Function to get the scale of the object in 3D space.
151 *
152 * @return The scale of the object as a Size3d struct.
153 */
154 virtual Size3d getScale() const { return {1.0f, 1.0f, 1.0f}; };
155 /**
156 * @brief Function to determine if the object can cast shadows. Can be
157 * overridden by derived classes.
158 *
159 * @return true if the object can cast shadows, false otherwise.
160 */
161 virtual bool canCastShadows() const { return false; };
162 virtual ~Renderable() = default;
163 /**
164 * @brief Function to determine if the object can use deferred rendering.
165 *
166 * @return (bool) True if the object supports deferred rendering, false
167 * otherwise.
168 */
169 virtual bool canUseDeferredRendering() { return true; };
170 virtual void beforePhysics() {}
171
172 /**
173 * @brief Whether the object should be included in depth of field
174 * calculations. When true, contributes to blur effects.
175 *
176 */
177 bool renderDepthOfView = false;
178
179 /**
180 * @brief Whether the object must be moved to the late forward rendering
181 * pipeline. Objects flagged here are skipped during the primary forward
182 * pass and rendered only after all standard forward elements.
183 */
184 bool renderLateForward = false;
185};
