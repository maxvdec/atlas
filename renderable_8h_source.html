22/**
23 * @brief An abstract base class representing any object that can be rendered in
24 * a Window. Contains virtual methods for rendering, initialization, updating,
25 * and setting view/projection matrices.
26 *
27 */
28class Renderable {
29 public:
30 /**
31 * @brief Function to render the object. Must be implemented by derived
32 * classes.
33 *
34 * @param dt Delta time since the last frame, useful for animations.
35 */
36 virtual void render(float dt) = 0;
37 /**
38 * @brief Function to initialize the object. Can be overridden by derived.
39 *
40 */
41 virtual void initialize() {};
42 /**
43 * @brief Function to update the object each frame.
44 * \warning It runs before the rendering phase and it should only contain
45 * logic updates.
46 *
47 * @param window The window where the object is being rendered.
48 */
49 virtual void update(Window &window) {};
50 /**
51 * @brief Function to set the view matrix for the object. Called from \ref
52 * Window for internal purposes.
53 *
54 * @param view The view matrix to set.
55 */
56 virtual void setViewMatrix(const glm::mat4 &view) {};
57 /**
58 * @brief Function to set the projection matrix for the object. Called from
59 * \ref Window for internal purposes.
60 *
61 * @param projection The projection matrix to set.
62 */
63 virtual void setProjectionMatrix(const glm::mat4 &projection) {};
64 /**
65 * @brief Function to get the current shader program used by the object.
66 *
67 * @return An optional containing the shader program if set, or \c
68 * std::nullopt if not.
69 */
70 virtual std::optional<ShaderProgram> getShaderProgram() {
71 return std::nullopt;
72 };
73 /**
74 * @brief Function to set the shader program for the object. Can be used to
75 * force an object to use a specific shader.
76 *
77 * @param shader The shader program to set.
78 */
79 virtual void setShader(const ShaderProgram &shader) {};
80 /**
81 * @brief Function to get the position of the object in 3D space.
82 *
83 * @return The position of the object as a Position3d struct.
84 */
85 virtual Position3d getPosition() const { return {0, 0, 0}; };
86 /**
87 * @brief Function to get the vertices of the object in 3D space.
88 *
89 * @return The vertices of the object as a vector of CoreVertex structs.
90 */
91 virtual std::vector<CoreVertex> getVertices() const { return {}; };
92 /**
93 * @brief Function to get the scale of the object in 3D space.
94 *
95 * @return The scale of the object as a Size3d struct.
96 */
97 virtual Size3d getScale() const { return {1, 1, 1}; };
98 /**
99 * @brief Function to determine if the object can cast shadows. Can be
100 * overridden by derived classes.
101 *
102 * @return true if the object can cast shadows, false otherwise.
103 */
104 virtual bool canCastShadows() const { return false; };
105 virtual ~Renderable() = default;
106};
