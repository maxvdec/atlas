293 void updateAutomaticAmbientFromSkybox() {
294 if (skybox != nullptr && skybox->cubemap.hasAverageColor) {
295 automaticAmbientColor = skybox->cubemap.averageColor;
296 double luminance = 0.2126 * automaticAmbientColor.r +
297 0.7152 * automaticAmbientColor.g +
298 0.0722 * automaticAmbientColor.b;
299 automaticAmbientIntensity =
300 static_cast<float>(std::clamp(luminance, 0.0, 1.0));
301 } else {
302 automaticAmbientColor = ambientLight.color;
303 automaticAmbientIntensity = ambientLight.intensity;
304 }
