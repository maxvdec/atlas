278 /**
279 * @brief Enables or disables using the atmosphere-generated skybox as the
280 * scene skybox.
281 *
282 * When enabled, the renderer is expected to populate `atmosphereSkybox`,
283 * and the scene will expose it via `getSkybox()`.
284 *
285 * When disabled, the scene will use the user-provided skybox set through
286 * `setSkybox()`.
287 */
288 void setUseAtmosphereSkybox(bool enabled) {
289 useAtmosphereSkybox = enabled;

291 if (automaticAmbient) {

293 }

356 void updateAutomaticAmbientFromSkybox() {
357 if (skybox != nullptr && skybox->cubemap.hasAverageColor) {
358 automaticAmbientColor = skybox->cubemap.averageColor;
359 double luminance = (0.2126 * automaticAmbientColor.r) +
360 (0.7152 * automaticAmbientColor.g) +
361 (0.0722 * automaticAmbientColor.b);
362 automaticAmbientIntensity =
363 static_cast<float>(std::clamp(luminance, 0.0, 1.0));
364 } else {
365 automaticAmbientColor = ambientLight.color;
366 automaticAmbientIntensity = ambientLight.intensity;
367 }
