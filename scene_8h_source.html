169 private:
170 std::vector<DirectionalLight *> directionalLights;
171 std::vector<Light *> pointLights;
172 std::vector<Spotlight *> spotlights;
173 Skybox *skybox = nullptr;
174 AmbientLight ambientLight = {{1.0f, 1.0f, 1.0f, 1.0f}, 0.5f / 4};
175 bool automaticAmbient = false;
176 Color automaticAmbientColor = Color::white();
177 float automaticAmbientIntensity = ambientLight.intensity;
178
179 void updateAutomaticAmbientFromSkybox() {
180 if (skybox != nullptr && skybox->cubemap.hasAverageColor) {
181 automaticAmbientColor = skybox->cubemap.averageColor;
182 double luminance = 0.2126 * automaticAmbientColor.r +
183 0.7152 * automaticAmbientColor.g +
184 0.0722 * automaticAmbientColor.b;
185 automaticAmbientIntensity =
186 static_cast<float>(std::clamp(luminance, 0.0, 1.0));
187 } else {
188 automaticAmbientColor = ambientLight.color;
189 automaticAmbientIntensity = ambientLight.intensity;
190 }
191 }
