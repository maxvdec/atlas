169/**
170 * @brief Structure representing a point in 3D space with additional data for
171 * collision detection.
172 *
173 */
174struct Point {
175 glm::vec3 xyz;
176 glm::vec3 ptA;
177 glm::vec3 ptB;
178
179 Point() : xyz(0.0f), ptA(0.0f), ptB(0.0f) {}
180
181 inline const Point &operator=(const Point &other) {
182 xyz = other.xyz;
183 ptA = other.ptA;
184 ptB = other.ptB;
185 return *this;
186 }
187
188 inline bool operator==(const Point &other) const {
189 return glm::all(glm::equal(xyz, other.xyz)) &&
190 glm::all(glm::equal(ptA, other.ptA)) &&
191 glm::all(glm::equal(ptB, other.ptB));
192 }
193};

429/**
430 * @brief Computes a cofactor from a 4x4 matrix.
431 *
432 * @param m The matrix to compute cofactor from.
433 * @param row Row index.
434 * @param col Column index.
435 * @return (float) The cofactor value.
436 */
437inline float takeCofactor(const glm::mat4 &m, int row, int col) {
438 glm::mat3 minor;
439 int r = 0;
440 for (int i = 0; i < 4; ++i) {
441 if (i == row)
442 continue;
443 int c = 0;
444 for (int j = 0; j < 4; ++j) {
445 if (j == col)
446 continue;
447 minor[r][c] = m[i][j];
448 ++c;
449 }
450 ++r;
451 }
452 float det = glm::determinant(minor);
453 return ((row + col) % 2 == 0 ? 1.0f : -1.0f) * det;
454}

456/**
457 * @brief Computes two orthogonal vectors from a given vector.
458 *
459 * @param v Input vector.
460 * @param u Output first orthogonal vector.
461 * @param w Output second orthogonal vector.
462 */
463inline void takeOrtho(const glm::vec3 &v, glm::vec3 &u, glm::vec3 &w) {
464 glm::vec3 n = glm::normalize(v);
465
466 glm::vec3 axis;
467 if (std::abs(n.x) < std::abs(n.y) && std::abs(n.x) < std::abs(n.z)) {
468 axis = glm::vec3(1.0f, 0.0f, 0.0f);
469 } else if (std::abs(n.y) < std::abs(n.z)) {
470 axis = glm::vec3(0.0f, 1.0f, 0.0f);
471 } else {
472 axis = glm::vec3(0.0f, 0.0f, 1.0f);
473 }
474
475 u = glm::normalize(glm::cross(n, axis));
476 w = glm::normalize(glm::cross(n, u));
477}
