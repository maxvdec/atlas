35/**
36 * @brief Describes the visual and environmental parameters for a terrain
37 * region.
38 *
39 * \subsection biome-example Example
40 * ```cpp
41 * Biome grassland("Grassland", Color(0.3, 0.6, 0.2, 1.0));
42 * grassland.minHeight = 0.0f;
43 * grassland.maxHeight = 20.0f;
44 * grassland.condition = [](Biome &b) {
45 * b.attachTexture(Texture::fromResourceName("Grass"));
46 * };
47 * ```
48 */
49class Biome {
50 public:
51 /**
52 * @brief Human readable name used for identification.
53 */
54 std::string name;
55 /**
56 * @brief Texture applied when `useTexture` is enabled.
57 */
58 Texture texture;
59 /**
60 * @brief Fallback tint applied when no texture is used.
61 */
62 Color color = {1.0, 1.0, 1.0, 1.0};
63
64 /**
65 * @brief Whether the biome should render the assigned texture.
66 */
67 bool useTexture = false;
68
69 /**
70 * @brief Assigns a texture and enables textured rendering for the biome.
71 *
72 * @param tex Texture resource to sample.
73 */
74 void attachTexture(const Texture &tex) {
75 this->texture = tex;
76 this->useTexture = true;
77 }
78
79 /**
80 * @brief Minimum height threshold for this biome to appear.
81 */
82 float minHeight = -1.0f;
83 /**
84 * @brief Maximum height threshold for this biome to appear.
85 */
86 float maxHeight = -1.0f;
87 /**
88 * @brief Minimum moisture value required for this biome.
89 */
90 float minMoisture = -1.0f;
91 /**
92 * @brief Maximum moisture value allowed for this biome.
93 */
94 float maxMoisture = -1.0f;
95 /**
96 * @brief Minimum temperature threshold.
97 */
98 float minTemperature = -1.0f;
99 /**
100 * @brief Maximum temperature threshold.
101 */
102 float maxTemperature = -1.0f;
103
104 /**
105 * @brief Creates a biome that uses both a texture and color tint.
106 *
107 * @param biomeName Name assigned to the biome.
108 * @param biomeTexture Texture used when sampling.
109 * @param biomeColor Fallback color tint.
110 */
111 Biome(const std::string &biomeName = "",
112 const Texture &biomeTexture = Texture(),
113 const Color &biomeColor = Color())
114 : name(biomeName), texture(biomeTexture), color(biomeColor) {}
115
116 /**
117 * @brief Creates a color-only biome configuration.
118 *
119 * @param biomeName Name assigned to the biome.
120 * @param biomeColor Color tint to apply.
121 */
122 Biome(const std::string &biomeName = "", const Color &biomeColor = Color())
123 : name(biomeName), color(biomeColor) {}
124
125 /**
126 * @brief Optional hook executed before the biome is rasterized.
127 */
128 BiomeFunction condition = [](Biome &thisBiome) { return; };
129};
