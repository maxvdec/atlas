36/**
37 * @brief Describes the visual and environmental parameters for a terrain
38 * region.
39 *
40 * \subsection biome-example Example
41 * ```cpp
42 * Biome grassland("Grassland", Color(0.3, 0.6, 0.2, 1.0));
43 * grassland.minHeight = 0.0f;
44 * grassland.maxHeight = 20.0f;
45 * grassland.condition = [](Biome &b) {
46 * b.attachTexture(Texture::fromResourceName("Grass"));
47 * };
48 * ```
49 */
50class Biome {
51 public:
52 /**
53 * @brief Human readable name used for identification.
54 */
55 std::string name;
56 /**
57 * @brief Texture applied when `useTexture` is enabled.
58 */
59 Texture texture;
60 /**
61 * @brief Fallback tint applied when no texture is used.
62 */
63 Color color = {1.0, 1.0, 1.0, 1.0};
64
65 /**
66 * @brief Whether the biome should render the assigned texture.
67 */
68 bool useTexture = false;
69
70 /**
71 * @brief Assigns a texture and enables textured rendering for the biome.
72 *
73 * @param tex Texture resource to sample.
74 */
75 void attachTexture(const Texture &tex) {
76 this->texture = tex;
77 this->useTexture = true;
78 }
79
80 /**
81 * @brief Minimum height threshold for this biome to appear.
82 */
83 float minHeight = -1.0f;
84 /**
85 * @brief Maximum height threshold for this biome to appear.
86 */
87 float maxHeight = -1.0f;
88 /**
89 * @brief Minimum moisture value required for this biome.
90 */
91 float minMoisture = -1.0f;
92 /**
93 * @brief Maximum moisture value allowed for this biome.
94 */
95 float maxMoisture = -1.0f;
96 /**
97 * @brief Minimum temperature threshold.
98 */
99 float minTemperature = -1.0f;
100 /**
101 * @brief Maximum temperature threshold.
102 */
103 float maxTemperature = -1.0f;

105 /**
106 * @brief Creates a biome that uses both a texture and color tint.
107 *
108 * @param biomeName Name assigned to the biome.
109 * @param biomeTexture Texture used when sampling.
110 * @param biomeColor Fallback color tint.
111 */
112 Biome(const std::string &biomeName = "",
113 const Texture &biomeTexture = Texture(),
114 const Color &biomeColor = Color())
115 : name(biomeName), texture(biomeTexture), color(biomeColor) {}
116
117 /**
118 * @brief Creates a color-only biome configuration.
119 *
120 * @param biomeName Name assigned to the biome.
121 * @param biomeColor Color tint to apply.
122 */
123 Biome(const std::string &biomeName = "", const Color &biomeColor = Color())
124 : name(biomeName), color(biomeColor) {}
125
126 /**
127 * @brief Optional hook executed before the biome is rasterized.
128 */
129 BiomeFunction condition = [](Biome &) { return; };
130};
