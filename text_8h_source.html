133/**
134 * @brief Represents a text object in the game world. It allows rendering of
135 * text with a specific font, position, and color.
136 * \subsection text-example Example
137 * ```cpp
138 * // Create a text object
139 * Text myText("Hello, Atlas!", Font::getFont("MyFont"), {100, 200},
140 * Color::white());
141 * // Set the position of the text
142 * myText.position = {150, 250};
143 * // Set the color of the text
144 * myText.color = Color(0.0f, 1.0f, 0.0f, 1.0f); // Green color
145 * // Add the text object to the scene
146 * scene.addObject(&myText);
147 * ```
148 */
149class Text : public UIObject {
150public:
151 /**
152 * @brief The content of the text to render.
153 *
154 */
155 std::string content;
156 /**
157 * @brief The font used to render the text.
158 *
159 */

163 *
164 */
165 Position2d position;
166 /**
167 * @brief The color of the text.
168 *
169 */
170 Color color = Color::white();
171 /**
172 * @brief Function that constructs a new Text object.
173 *
174 */
175 Text() {
176 };
177 /**
178 * @brief Function that constructs a new Text object with the given
179 * parameters.
180 *
181 * @param text The text content to display.
182 * @param font The font with which to render the text.
183 * @param position The position of the text.
184 * @param color The color of the text.
185 */
186 Text(const std::string& text, const Font& font,
187 Position2d position = {0, 0}, const Color& color = Color::white())

202
203private:
204 std::shared_ptr<opal::DrawingState> vao = nullptr;
205 std::shared_ptr<opal::Buffer> vertexBuffer = nullptr;
206 size_t vertexBufferCapacity = sizeof(float) * 6 * 4; // capacity in bytes
207 glm::mat4 projection;
208 ShaderProgram shader;
209};
