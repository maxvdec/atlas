85/**
86 * @brief Structure that holds the data for a texture. It is created from a
87 * resource.
88 * \subsection texture-example Example
89 * ```cpp
90 * // Load a texture from a resource
91 * Texture myTexture = Texture::fromResourceName("MyTextureResource",
92 * TextureType::Color);
93 * // Create a checkerboard texture
94 * Texture checkerTexture = Texture::createCheckerboard(512, 512, 32,
95 * Color::white(), Color::black());
96 * // Create a tiled checkerboard texture
97 * std::vector<CheckerTile> tiles = {
98 * {Color::red(), Color::black(), 64},
99 * {Color::green(), Color::black(), 32},
100 * {Color::blue(), Color::black(), 16}
101 * };
102 * Texture tiledTexture = Texture::createTiledCheckerboard(512, 512, tiles);
103 * ```
104 *
105 */

120 * @brief The core Id for the texture in OpenGL.

124 /**
125 * @brief The type of the texture (e.g., Color, Specular, Cubemap).
126 *
127 */
128 TextureType type;

281 * // Show the render target again

283 * ```
284 *
285 */
