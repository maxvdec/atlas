45/**
46 * @brief Enumeration of texture wrapping modes. These modes determine how
47 * textures are sampled when texture coordinates fall outside the standard range
48 * of [0, 1].
49 *
50 */
51enum class TextureWrappingMode {
52 /**
53 * @brief Repeats the texture infinitely in both directions.
54 */
55 Repeat,
56 /**
57 * @brief Alternates the texture direction on every repeat for a mirrored
58 * look.
59 */
60 MirroredRepeat,
61 /**
62 * @brief Clamps coordinates to the texture edge, stretching edge pixels.
63 */
64 ClampToEdge,
65 /**
66 * @brief Uses a constant border color outside the [0,1] range.
67 */
68 ClampToBorder
69};

71/**
72 * @brief Enumeration of texture filtering modes. These modes determine how
73 * textures are sampled when they are minified or magnified.

75 */
76enum class TextureFilteringMode {
77 /**
78 * @brief Picks the texel closest to the lookup coordinate.

80 Nearest,
81 /**
82 * @brief Linearly interpolates texels for smoother sampling.

84 Linear

140/**
