142 * @brief Structure representing rotation in 3D space using Euler angles.
143 * Provides arithmetic operations and conversions to/from GLM quaternions.
144 *
145 * \subsection rotation3d-example Example
146 * ```cpp
147 * // Create a rotation (pitch, yaw, roll in degrees)
148 * Rotation3d rot(0.0f, 45.0f, 0.0f);
149 * // Rotate object over time

155struct Rotation3d {
156 float pitch; // Rotation around the X-axis
157 float yaw; // Rotation around the Y-axis
158 float roll; // Rotation around the Z-axis
159
160 Rotation3d() : pitch(0.0f), yaw(0.0f), roll(0.0f) {}
161
162 Rotation3d(float pitch, float yaw, float roll)
163 : pitch(pitch), yaw(yaw), roll(roll) {}

165 Rotation3d(double pitch, double yaw, double roll)
166 : pitch(static_cast<float>(pitch)), yaw(static_cast<float>(yaw)),
167 roll(static_cast<float>(roll)) {}

169 Rotation3d operator+(const Rotation3d &other) const {
170 return {pitch + other.pitch, yaw + other.yaw, roll + other.roll};
171 }
172
173 Rotation3d operator-(const Rotation3d &other) const {
174 return {pitch - other.pitch, yaw - other.yaw, roll - other.roll};
175 }
176
177 Rotation3d operator*(float scalar) const {
178 return {pitch * scalar, yaw * scalar, roll * scalar};
179 }

181 bool operator==(const Rotation3d &other) const {
182 return pitch == other.pitch && yaw == other.yaw && roll == other.roll;
183 }
184

186 return {pitch / scalar, yaw / scalar, roll / scalar};
187 }

189 inline glm::vec3 toGlm() const { return glm::vec3(pitch, yaw, roll); }
190
191 inline glm::quat toGlmQuat() const {
192 const float pitchRad = glm::radians(pitch);
193 const float yawRad = glm::radians(yaw);
194 const float rollRad = glm::radians(roll);
195
196 const glm::quat qYaw = glm::angleAxis(yawRad, glm::vec3(0, 1, 0));
197 const glm::quat qPitch = glm::angleAxis(pitchRad, glm::vec3(1, 0, 0));
198 const glm::quat qRoll = glm::angleAxis(rollRad, glm::vec3(0, 0, 1));
199
200 return qRoll * qPitch * qYaw;
201 }
202
203 inline static Rotation3d fromGlmQuat(const glm::quat &quat) {
204 glm::mat3 m = glm::mat3_cast(quat);
205
206 float sPitch = glm::clamp(m[1][2], -1.0f, 1.0f);
207 float pitchRad = std::asin(sPitch);
208 float cPitch = std::cos(pitchRad);
209
210 float yawRad;
211 float rollRad;
212
213 constexpr float eps = 1e-6f;
214 if (std::abs(cPitch) > eps) {
215 yawRad = std::atan2(-m[0][2], m[2][2]);
216 rollRad = std::atan2(-m[1][0], m[1][1]);
217 } else {
218 yawRad = std::atan2(sPitch * m[0][1], m[0][0]);
219 rollRad = 0.0f;
220 }
221
222 return {glm::degrees(pitchRad), glm::degrees(yawRad),
223 glm::degrees(rollRad)};
224 }
225
226 inline static Rotation3d fromGlm(const glm::vec3 &vec) {
227 return {vec.x, vec.y, vec.z};
228 }
