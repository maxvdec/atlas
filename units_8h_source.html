151struct BoundingBox {
152 /** @brief Minimum corner of the box (inclusive). */
153 Position3d min;
154 /** @brief Maximum corner of the box (inclusive). */
155 Position3d max;
156
157 BoundingBox() : min(Position3d::zero()), max(Position3d::zero()) {}
158 BoundingBox(const Position3d &min, const Position3d &max)
159 : min(min), max(max) {}
160
161 /** @brief Returns true when the point lies inside the bounds. */
162 bool contains(const Position3d &point) const {
163 return (point.x >= min.x && point.x <= max.x) &&
164 (point.y >= min.y && point.y <= max.y) &&
165 (point.z >= min.z && point.z <= max.z);
166 }
167
168 /** @brief Returns true when two axis-aligned boxes overlap. */
169 bool intersects(const BoundingBox &other) const {
170 return (min.x <= other.max.x && max.x >= other.min.x) &&
171 (min.y <= other.max.y && max.y >= other.min.y) &&
172 (min.z <= other.max.z && max.z >= other.min.z);
173 }

176/**
177 * @brief Structure representing rotation in 3D space using Euler angles.
178 * Provides arithmetic operations and conversions to/from GLM quaternions.
179 *
180 * \subsection rotation3d-example Example

182 * // Create a rotation (pitch, yaw, roll in degrees)

190struct Rotation3d {
191 float pitch; // Rotation around the X-axis
192 float yaw; // Rotation around the Y-axis

194
195 Rotation3d() : pitch(0.0f), yaw(0.0f), roll(0.0f) {}

197 Rotation3d(float pitch, float yaw, float roll)
198 : pitch(pitch), yaw(yaw), roll(roll) {}

200 Rotation3d(double pitch, double yaw, double roll)

202 roll(static_cast<float>(roll)) {}

204 Rotation3d operator+(const Rotation3d &other) const {
205 return {pitch + other.pitch, yaw + other.yaw, roll + other.roll};
206 }

208 Rotation3d operator-(const Rotation3d &other) const {
209 return {pitch - other.pitch, yaw - other.yaw, roll - other.roll};
210 }
211

213 return {pitch * scalar, yaw * scalar, roll * scalar};
214 }

217 return pitch == other.pitch && yaw == other.yaw && roll == other.roll;
218 }
219
220 Rotation3d operator/(float scalar) const {

222 }

224 glm::vec3 toGlm() const { return glm::vec3(pitch, yaw, roll); }
225
226 glm::quat toGlmQuat() const {
227 const float pitchRad = glm::radians(pitch);
228 const float yawRad = glm::radians(yaw);
229 const float rollRad = glm::radians(roll);
230
231 const glm::quat qYaw = glm::angleAxis(yawRad, glm::vec3(0, 1, 0));
232 const glm::quat qPitch = glm::angleAxis(pitchRad, glm::vec3(1, 0, 0));
233 const glm::quat qRoll = glm::angleAxis(rollRad, glm::vec3(0, 0, 1));
234
235 return qRoll * qPitch * qYaw;
236 }
237
238 static Rotation3d fromGlmQuat(const glm::quat &quat) {
239 glm::mat3 m = glm::mat3_cast(quat);
240
241 float sPitch = glm::clamp(m[1][2], -1.0f, 1.0f);
242 float pitchRad = std::asin(sPitch);
243 float cPitch = std::cos(pitchRad);
244
245 float yawRad;
246 float rollRad;
247
248 constexpr float eps = 1e-6f;
249 if (std::abs(cPitch) > eps) {
250 yawRad = std::atan2(-m[0][2], m[2][2]);
251 rollRad = std::atan2(-m[1][0], m[1][1]);
252 } else {
253 yawRad = std::atan2(sPitch * m[0][1], m[0][0]);
254 rollRad = 0.0f;
255 }
256
257 return {glm::degrees(pitchRad), glm::degrees(yawRad),
258 glm::degrees(rollRad)};
259 }
260
261 static Rotation3d fromGlm(const glm::vec3 &vec) {
262 return {vec.x, vec.y, vec.z};
263 }
