19 * @brief Structure representing a position in 3D space with double precision.
20 * Provides arithmetic operations and conversions to/from GLM types.
21 *
22 */
23struct Position3d {
24 double x;
25 double y;
26 double z;
27
28 Position3d operator+(const Position3d &other) const {
29 return {x + other.x, y + other.y, z + other.z};
30 }
31
32 Position3d operator-(const Position3d &other) const {
33 return {x - other.x, y - other.y, z - other.z};
34 }
35
36 Position3d operator*(double scalar) const {
37 return {x * scalar, y * scalar, z * scalar};
38 }
39
40 Position3d operator/(double scalar) const {
41 return {x / scalar, y / scalar, z / scalar};
42 }
43
44 inline glm::vec3 toGlm() const {
45 return glm::vec3(static_cast<float>(x), static_cast<float>(y),
46 static_cast<float>(z));
47 }
48
49 Position3d operator+(const glm::vec3 &vec) const {
50 return {x + vec.x, y + vec.y, z + vec.z};
51 }
52
53 Position3d operator-(const glm::vec3 &vec) const {
54 return {x - vec.x, y - vec.y, z - vec.z};
55 }
56
57 void operator+=(const Position3d &vec) {
58 x += vec.x;
59 y += vec.y;
60 z += vec.z;
61 }
62
63 void operator-=(const Position3d &vec) {
64 x -= vec.x;
65 y -= vec.y;
66 z -= vec.z;
67 }
68
69 inline Position3d normalized() const {
70 double length = std::sqrt(x * x + y * y + z * z);
71 if (length == 0)
72 return {0, 0, 0};
73 return {x / length, y / length, z / length};
74 }
75
76 inline static Position3d fromGlm(const glm::vec3 &vec) {
77 return {static_cast<double>(vec.x), static_cast<double>(vec.y),
78 static_cast<double>(vec.z)};
79 }
80
81 inline friend std::ostream &operator<<(std::ostream &os,
82 const Position3d &p) {
83 os << "Position3d(" << p.x << ", " << p.y << ", " << p.z << ")";
84 return os;
85 }
86};

129 return {pitch - other.pitch, yaw - other.yaw, roll - other.roll};
130 }
131
132 Rotation3d operator*(double scalar) const {
133 return {pitch * scalar, yaw * scalar, roll * scalar};
134 }
135
136 Rotation3d operator/(double scalar) const {
137 return {pitch / scalar, yaw / scalar, roll / scalar};
138 }
139
140 inline glm::vec3 toGlm() const {
141 return glm::vec3(static_cast<float>(pitch), static_cast<float>(yaw),
142 static_cast<float>(roll));
143 }
144
145 inline glm::quat toGlmQuat() const {
146 glm::vec3 eulerAngles = toGlm();
147 glm::vec3 radians = glm::radians(eulerAngles);
148 return glm::quat(radians);
149 }
150
151 inline static Rotation3d fromGlmQuat(const glm::quat &quat) {
152 glm::vec3 euler = glm::degrees(glm::eulerAngles(quat));
153 return {static_cast<double>(euler.x), static_cast<double>(euler.y),
154 static_cast<double>(euler.z)};
155 }

158 return {static_cast<double>(vec.x), static_cast<double>(vec.y),
