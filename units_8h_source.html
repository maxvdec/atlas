23struct Position3d {
24 double x;
25 double y;
26 double z;
27
28 Position3d operator+(const Position3d &other) const {
29 return {x + other.x, y + other.y, z + other.z};
30 }
31
32 Position3d operator-(const Position3d &other) const {
33 return {x - other.x, y - other.y, z - other.z};
34 }
35
36 Position3d operator*(double scalar) const {
37 return {x * scalar, y * scalar, z * scalar};
38 }
39
40 Position3d operator/(double scalar) const {
41 return {x / scalar, y / scalar, z / scalar};
42 }
43
44 bool operator==(const Position3d &other) const {
45 return x == other.x && y == other.y && z == other.z;
46 }
47
48 inline glm::vec3 toGlm() const {
49 return glm::vec3(static_cast<float>(x), static_cast<float>(y),
50 static_cast<float>(z));
51 }
52
53 Position3d operator+(const glm::vec3 &vec) const {
54 return {x + vec.x, y + vec.y, z + vec.z};
55 }
56
57 Position3d operator-(const glm::vec3 &vec) const {
58 return {x - vec.x, y - vec.y, z - vec.z};
59 }
60
61 void operator+=(const Position3d &vec) {
62 x += vec.x;
63 y += vec.y;
64 z += vec.z;
65 }
66
67 void operator-=(const Position3d &vec) {
68 x -= vec.x;
69 y -= vec.y;
70 z -= vec.z;
71 }
72
73 inline Position3d normalized() const {
74 double length = std::sqrt(x * x + y * y + z * z);
75 if (length == 0)
76 return {0, 0, 0};
77 return {x / length, y / length, z / length};
78 }
79
80 inline static Position3d fromGlm(const glm::vec3 &vec) {
81 return {static_cast<double>(vec.x), static_cast<double>(vec.y),
82 static_cast<double>(vec.z)};
83 }
