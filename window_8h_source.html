48struct WindowConfiguration {
49 /**
50 * @brief The title of the window.
51 *
52 */
53 std::string title;
54 /**
55 * @brief The width of the window in pixels.
56 *
57 */
58 int width;

60 * @brief The height of the window in pixels.
61 *
62 */
63 int height;
64 /**
65 * @brief Internal rendering resolution scale factor. Values below 1.0
66 * render at a lower resolution and upscale to the window, improving
67 * performance.
68 */
69 float renderScale = 0.75f;
70 /**
71 * @brief Whether the mouse cursor should be captured and hidden.
72 *
73 */
74 bool mouseCaptured = true;
75 /**
76 * @brief The X position of the window. Use WINDOW_CENTERED to center.
77 *
78 */
79 int posX = WINDOW_CENTERED;
80 /**
81 * @brief The Y position of the window. Use WINDOW_CENTERED to center.
82 *
83 */
84 int posY = WINDOW_CENTERED;
85 /**
86 * @brief Whether to enable multisampling anti-aliasing.
87 *
88 */
89 bool multisampling = true;
90 /**
91 * @brief Whether the window should have decorations (title bar, borders).
92 *
93 */
94 bool decorations = true;
95 /**
96 * @brief Whether the window can be resized by the user.
97 *
98 */
99 bool resizable = true;
100 /**
101 * @brief Whether the window framebuffer should be transparent.
102 *
103 */
104 bool transparent = false;
105 /**
106 * @brief Whether the window should always stay on top of other windows.
107 *
108 */
109 bool alwaysOnTop = false;
110 /**
111 * @brief The opacity of the window. 1.0 is fully opaque, 0.0 is fully
112 * transparent.
113 *
114 */
115 float opacity = 1.0f;
116 /**
117 * @brief The width component of the aspect ratio. Use -1 for no
118 * constraint.
119 *
120 */
121 int aspectRatioX = -1;
122 /**
123 * @brief The height component of the aspect ratio. Use -1 for no
124 * constraint.
125 *
126 */
127 int aspectRatioY = -1;
128 /**
129 * @brief Resolution scale used specifically for SSAO passes. Lower values
130 * reduce quality but greatly boost performance.
131 */
132 float ssaoScale = 0.5f;
133};
